114 void AHX_Thread(void* param)
115 {
116  // int i;
117  iop_sema_t sema; // semaphore
118 
119  // set sema properties and create...
120  sema.attr = 0; // SA_THFIFO;
121  sema.initial = 0;
122  sema.max = 1;
123  transfer_sema= CreateSema(&sema);
124 
125  // check sema created successfully, or exit
126  if(transfer_sema <= 0) {
127  #ifndef COMPACT_CODE
128  M_PRINTF("FATAL - Failed to create semaphore!\n");
129  #endif
130  ExitDeleteThread();
131  }
132 
133  // allocate memory for SPU2 xfer buffer
134  spubuf = AllocSysMemory(0, 0x800, NULL); // 2048 bytes (enough for 2 SPU cycles at 512 bytes per channel)
135  pcmbuf = AllocSysMemory(0, 0xF00*8, NULL); // enough to hold 8 full AHX-PCM decoded chunks (0xF00 per chunk)
136  if(spubuf == NULL) {
137  #ifndef COMPACT_CODE
138  M_PRINTF("FATAL - Failed to allocate memory for sound buffer!\n");
139  #endif
140  ExitDeleteThread();
141  }
142 
143  // print version #
144  #ifndef COMPACT_CODE
145  M_PRINTF("AHXplayer %s Started\n", MODVERSION);
146  #endif
147 
148  AHX_ClearSoundBuffers();
149 
150  #ifndef COMPACT_CODE
151  M_PRINTF("Memory Allocated. %d bytes left.\n",QueryTotalFreeMemSize());
152  #endif
153 
154  // Init the RPC server
155  SifInitRpc(0);
156  SifSetRpcQueue(&qd, GetThreadId());
157  SifRegisterRpc(&Sd0, AHX_IRX, (void*)AHX_rpc_server, (void*)&buffer[0], 0, 0, &qd);

168 void AHX_PlayThread(void* param)
169 {
170  int i;
171  int chunk; // we transfer to 2 blocks of SPU mem, are we working chunk 1 or 2?
172 
173  // reset position and flags etc
174  AHX_ResetPlayThread();
175 
176  // main thread loop
177  while(1)
178  {
179  // backfill chunks that have already been passed to SPU. We do this before the WaitSema
180  // because we could be stuck there waiting for a while. If any spare time exists, we use
181  // it to mix the buffers...
182  if (playing)
183  {
184  for (i=0; i<7; i++)
185  {
186  // check if we've just passed a section of data and refill it
187  if (readpos>=0xF00*(i+1) && readpos<=0xF00*(i+2) && !seg_done[i])
188  {
189  if (playing)
190  {
191  AHXOutput_MixBuffer( (short*)(pcmbuf+(0xF00*i)) ); // remix
192  }
193  seg_done[i] = 1; // mark segment as mixed and ready
194  if (i>0) seg_done[i-1] = 0; else seg_done[7] = 0; // catch end segment
195  }
196  }
197  }
198 
199  // wait for SPU2 to set sema to indicate it's ready for more data
200  WaitSema(transfer_sema);
201 
202  // suspend interrupts so our sema trigger doesn't get fired more than once
203  // while we work in here...
204  CpuSuspendIntr(&intr_state);
205 
206  if (playing)
207  {
208  // get SPU transfer status to determine which area of SPU buffer to write PCM data to
209  chunk = 1 - (sceSdBlockTransStatus(1, 0 )>>24);
210 
211  wmemcpy(spubuf+(1024*chunk),pcmbuf+readpos,512); // left channel
212  wmemcpy(spubuf+(1024*chunk)+512,pcmbuf+readpos,512); // right channel (same cos we're mono)
213 
214  // SPU2 reads 512 bytes for each channel per frame, we're using the same PCM buffer for each
215  // channel (left/right), so update read position by 512 bytes. SPU reads data like so:
216  // 512 bytes for left channel | 512 bytes for right channel | 512 for left... etc etc
217  readpos += 512;
218 
219  // check to see whether we've reached the end of our PCM buffer, if so, reset read position
220  // remix end chunk of AHX data in PCM buffer and mark segment as complete
221  if(readpos >= 0xF00 * 8)
222  {
223  readpos = 0;
224  AHXOutput_MixBuffer( (short*)(pcmbuf+(0xF00*7)) );
225  seg_done[7] = 1;
226  seg_done[6] = 0;
227  }
228  }
229 
230  // re-enable interrupts
231  CpuResumeIntr(intr_state);
232  }
233 }
