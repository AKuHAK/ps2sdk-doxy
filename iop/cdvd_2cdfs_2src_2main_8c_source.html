141 {

153  DPRINTF(" buffer...... 0x%X\n", (int)buffer);
154  DPRINTF(" size........ %d\n\n", size);
155 
156  i = (int)f->privdata;
157 
158  if (i >= MAX_FILES_OPENED) {
159  printf("fio_read: ERROR: File does not appear to be open!\n");
160  return -1;
161  }
162 
163  // A few sanity checks
164  if (fd_table[i].filePos > fd_table[i].fileSize) {
165  // We cant start reading from past the beginning of the file
166  return 0; // File exists but we couldnt read anything from it
167  }
168 
169  if ((fd_table[i].filePos + size) > fd_table[i].fileSize)
170  size = fd_table[i].fileSize - fd_table[i].filePos;
171 
172  if (size <= 0)
173  return 0;
174 
175  if (size > MAX_BYTES_READ)
176  size = MAX_BYTES_READ;
177 
178  // Now work out where we want to start reading from
179  start_sector = fd_table[i].LBA + (fd_table[i].filePos >> 11);
180  off_sector = (fd_table[i].filePos & 0x7FF);
181 
182  num_sectors = (off_sector + size);
183  num_sectors = (num_sectors >> 11) + ((num_sectors & 2047) != 0);
184 
185  DPRINTF("fio_read: read sectors %d to %d\n", start_sector, start_sector + num_sectors);
186 
187  // Skip a Sector for equal (use the last sector in buffer)
188  if (start_sector == lastsector) {
189  read = 1;
190  if (last_bk > 0)
191  memcpy(local_buffer, local_buffer + 2048 * (last_bk), 2048);
192  last_bk = 0;
193  }
194 
195  lastsector = start_sector + num_sectors - 1;
196  // Read the data (we only ever get 16KB max request at once)
197 
198  if (read == 0 || (read == 1 && num_sectors > 1)) {
199  if (!cdfs_readSect(start_sector + read, num_sectors - read, local_buffer + ((read) << 11))) {
200  DPRINTF("Couldn't Read from file for some reason\n");
201  }
202 
203  last_bk = num_sectors - 1;
204  }
205 
206  memcpy(buffer, local_buffer + off_sector, size);
207  fd_table[i].filePos += size;
208 
209  return (size);
210 }
