132 int TransInterrupt(void *data)
133 {
134  IntrData *intr = (IntrData*) data;
135  s32 dir;
136  s32 core;
137 
138  dir = ((intr->mode & 0x200) < 1);
139  core = intr->mode & 0xFF;
140 
141  // Voice Transfer
142  if((intr->mode & 0x100) == 0)
143  {
144  // SD_C_STATX(core)
145  // If done elsewise, it doesn't work, havn't figured out why yet.
146  volatile u16 *statx = U16_REGISTER(0x344 + (core * 1024));
147 
148  if(!(*statx & 0x80)) while(!(*statx & 0x80));
149 
150  *SD_CORE_ATTR(core) &= ~SD_CORE_DMA;
151 
152  if(*SD_CORE_ATTR(core) & 0x30) while((*SD_CORE_ATTR(core) & 0x30));
153 
154  if(TransIntrHandlers[core]) goto intr_handler;
155  if(TransIntrCallbacks[core]) goto SdIntrCallback;
156 
157  VoiceTransComplete[core] = 1;
158  }
159  else
160  { // Block Transfer
161  if(intr->mode & (SD_TRANS_LOOP << 8))
162  {
163  // Switch buffers
164  BlockTransBuff[core] = 1 - BlockTransBuff[core];
165  // Setup DMA & send
166  *SD_DMA_ADDR(core) = (BlockTransSize[core] * BlockTransBuff[core])+BlockTransAddr[core];
167  *SD_DMA_SIZE(core) = (BlockTransSize[core]/64)+((BlockTransSize[core]&63)>0);
168  *SD_DMA_CHCR(core) = SD_DMA_START | SD_DMA_CS | dir;
169  }
170  else
171  {
172  *SD_CORE_ATTR(core) &= ~SD_CORE_DMA;
173  *SD_P_MMIX(core) &= 0xFF3F;
174  *U16_REGISTER(0x1B0 + (core * 1024)) = 0;
175  }
176 
177  if(TransIntrHandlers[core])
178  {
179  intr_handler:
180  TransIntrHandlers[core](core, intr->data);
181  }
182  else
183  {
184  if(TransIntrCallbacks[core])
185  {
186  SdIntrCallback:
187  TransIntrCallbacks[core](0);
188  }
189  }
190  }
191 
192  if(dir == SD_DMA_DIR_SPU2IOP) FlushDcache();
193 
194  return 1;
195 }

197 int Spu2Interrupt(void *data)
198 {
199  u16 val;
200 
201  val = ((*SD_C_IRQINFO)&0xc)>>2;
202  if (!val)
203  return 1;
204 
205  if (val&1)
206  *SD_CORE_ATTR(0) = *SD_CORE_ATTR(0) & 0xffbf;
207 
208  if (val&2)
209  *SD_CORE_ATTR(1) = *SD_CORE_ATTR(1) & 0xffbf;
210 
211  if(Spu2IntrHandler != NULL)
212  {
213  Spu2IntrHandler(val, Spu2IntrData);
214  }
215  else
216  {
217  if(Spu2IrqCallback) Spu2IrqCallback(0);
218  }
219 
220  return 1;
221 }

341 void InitVoices()
342 {
343  s32 voice, i;
344  volatile u16 *statx;
345 
346  // Set Start Address of data to transfer.
347  *SD_A_TSA_HI(0) = 0;
348  *SD_A_TSA_LO(0) = 0x5000 >> 1;
349 
350  // Fill with data.
351  // First 16 bytes are reserved.
352  for(i = 0; i < 16; i++) *SD_A_STD(0) = VoiceDataInit[i];
353 
354  // Set Transfer mode to IO
355  *SD_CORE_ATTR(0) = (*SD_CORE_ATTR(0) & ~SD_CORE_DMA) | SD_DMA_IO;
356 
357  statx = U16_REGISTER(0x344);
358 
359  // Wait for transfer to complete;
360  while(*statx & SD_IO_IN_PROCESS);
361 
362  // Reset DMA settings
363  *SD_CORE_ATTR(0) &= ~SD_CORE_DMA;
364 
365  // Init voices
366  for(voice = 0; voice < 24; voice++)
367  {
368  *SD_VP_VOLL(0, voice) = 0;
369  *SD_VP_VOLR(0, voice) = 0;
370  *SD_VP_PITCH(0, voice) = 0x3FFF;
371  *SD_VP_ADSR1(0, voice) = 0;
372  *SD_VP_ADSR2(0, voice) = 0;
373 
374  *SD_VP_VOLL(1, voice) = 0;
375  *SD_VP_VOLR(1, voice) = 0;
376  *SD_VP_PITCH(1, voice) = 0x3FFF;
377  *SD_VP_ADSR1(1, voice) = 0;
378  *SD_VP_ADSR2(1, voice) = 0;
379 
380  // Top address of waveform data
381  *SD_VA_SSA_HI(0, voice) = 0;
382  *SD_VA_SSA_LO(0, voice) = 0x5000 >> 1;
383  *SD_VA_SSA_HI(1, voice) = 0;
384  *SD_VA_SSA_LO(1, voice) = 0x5000 >> 1;
385  }
386 
387  // Set all voices to ON
388  *SD_A_KON_HI(0) = 0xFFFF;
389  *SD_A_KON_LO(0) = 0xFF;
390  *SD_A_KON_HI(1) = 0xFFFF;
391  *SD_A_KON_LO(1) = 0xFF;
392 
393  // There is no guarantee that voices will be turn on at once.
394  // So we wait to make sure.
395  nopdelay();
396 
397  // Set all voices to OFF
398  *SD_A_KOFF_HI(0) = 0xFFFF;
399  *SD_A_KOFF_LO(0) = 0xFF;
400  *SD_A_KOFF_HI(1) = 0xFFFF;
401  *SD_A_KOFF_LO(1) = 0xFF;
402 
403  // There is no guarantee that voices will be turn off at once.
404  // So we wait to make sure.
405  nopdelay();
406 
407  *SD_S_ENDX_HI(0) = 0;
408  *SD_S_ENDX_LO(0) = 0;
409 }
