86  return;
87 
88  memPool.hcIsoTdToIoReqLUT[arg - memPool.hcIsoTdBuf] = NULL;

90  req->transferedBytes = 0;
91  req->resultCode = (pswRes << 4) | tdHcRes;
92 
93  if ((tdHcRes == USB_RC_OK) && ((pswRes == USB_RC_OK) || (pswRes == USB_RC_DATAUNDER))) {
94  if ((req->correspEndpoint->hcEd.hcArea & HCED_DIR_MASK) == HCED_DIR_IN)
95  req->transferedBytes = pswOfs;
96  else
97  req->transferedBytes = req->length;
98  }
99 
100  req->prev = listEnd;
101  if (listEnd)
102  listEnd->next = req;
103  else
104  listStart = req;
105  req->next = NULL;
106  listEnd = req;
107 
108  HcED *ed = &req->correspEndpoint->hcEd;
109  if (ED_HALTED(req->correspEndpoint->hcEd)) {
110  HcIsoTD *curTd = (HcIsoTD *)((u32)ed->tdHead & ~0xF);
111 
112  while (curTd && (curTd != (HcIsoTD*)ed->tdTail)) {
113  HcIsoTD *nextTd = curTd->next;
114  freeIsoTd(curTd);
115 
116  IoRequest *req = memPool.hcIsoTdToIoReqLUT[curTd - memPool.hcIsoTdBuf];
117  if (req) {
118  memPool.hcIsoTdToIoReqLUT[arg - memPool.hcIsoTdBuf] = NULL;
119  IoRequest *listPos;
120  for (listPos = listStart; listPos != NULL; listPos = listPos->next)
121  if (listPos == req)
122  break;
123  if (listPos == NULL) {
124  req->resultCode = USB_RC_ABORTED;
125  req->prev = listEnd;
126  if (listEnd)
127  listEnd->next = req;
128  else
129  listStart = req;
130  req->next = NULL;
131  listEnd = req;
132  }
133  }
134  curTd = nextTd;
135  }
136  ed->tdHead = ed->tdTail;
137  }
138 
139  IoRequest *listPos = listStart;
140  while (listPos) {
141  IoRequest *listNext = listPos->next;
142  listPos->busyFlag = 0;
143  if (listPos->correspEndpoint->correspDevice) {
144  if (listPos->callbackProc)
145  listPos->callbackProc(listPos);
146  } else
147  freeIoRequest(listPos);
148  listPos = listNext;
149  }
150  checkTdQueue(ISOTD_QUEUE);

153 void processDoneQueue_GenTd(HcTD *arg) {
154  IoRequest *req;
155  IoRequest *firstElem = NULL, *lastElem = NULL;
156 
157  u32 hcRes;
158 
159  if ((req = memPool.hcTdToIoReqLUT[arg - memPool.hcTdBuf])) {
160  memPool.hcTdToIoReqLUT[arg - memPool.hcTdBuf] = NULL;
161 
162  u32 tdHcArea = arg->HcArea;
163 
164  if (arg->bufferEnd && (tdHcArea & 0x180000)) { // dir != SETUP
165  if (arg->curBufPtr == 0) // transfer successful
166  req->transferedBytes = req->length;
167  else
168  req->transferedBytes = (u8 *)arg->curBufPtr - (u8 *)req->destPtr;
169  }
170  hcRes = tdHcArea >> 28;
171  freeTd(arg);
172 
173  if (req->resultCode == USB_RC_OK)
174  req->resultCode = hcRes;
175 
176  if (hcRes || ((tdHcArea & 0xE00000) != 0xE00000)) { // E00000: interrupts disabled
177  req->prev = lastElem;
178  if (lastElem)
179  lastElem->next = req;
180  else
181  firstElem = req;
182  req->next = NULL;
183  lastElem = req;
184  }
185 
186  HcED *ed = &req->correspEndpoint->hcEd;
187  if (hcRes && ED_HALTED(req->correspEndpoint->hcEd)) {
188  HcTD *tdListPos = (HcTD*)((u32)ed->tdHead & ~0xF);
189  while (tdListPos && (tdListPos != ed->tdTail)) {
190  HcTD *nextTd = tdListPos->next;
191  freeTd(tdListPos);
192 
193  IoRequest *req = memPool.hcTdToIoReqLUT[tdListPos - memPool.hcTdBuf];
194  if (req) {

199  if (listPos == req)
