153  }
154

156  fileInode=pfsInodeGetFileInDir(parentInode, file, &result);

170  result=-EEXIST;
171  }

174  count = 0;

186  file, &result);

191  }

207  (openFlags & O_TRUNC))

251  cached->flags |= PFS_CACHE_FLAG_DIRTY;
252  pfsCacheFree(cached);

254  result=result3;

256  }else{

258  PFS_CACHE_FLAG_NOLOAD | PFS_CACHE_FLAG_NOTHING, &result4);
259  if (cached)
260  {

262  cached->u.aentry->aLen=sizeof(pfs_inode_t);

275  }
276  }
277  }

283  if (openFlags & O_APPEND)
284  freeSlot->position = fileInode->u.inode->size;
285  else

295  if (pfsMount->flags & PFS_FIO_ATTR_WRITEABLE)

299  return 0;

301  freeSlot->clink=NULL;
302  }
303

305 
306  return pfsFioCheckForLastError(pfsMount, result);

309 static int fileTransferRemainder(pfs_file_slot_t *fileSlot, void *buf, int size, int operation)
310 {
311  u32 sector, pos;
312  s32 i;
313  int result;
314  pfs_blockpos_t *blockpos = &fileSlot->block_pos;
315  pfs_mount_t *pfsMount = fileSlot->clink->pfsMount;
316  pfs_unaligned_io_t *unaligned = &fileSlot->unaligned;
317  pfs_unaligned_io_t *unaligned2 = NULL;
318  pfs_blockinfo_t *bi = pfsBlockGetCurrent(blockpos);
319 
320  sector = ((bi->number+blockpos->block_offset) << pfsMount->sector_scale) +
321  (blockpos->byte_offset >> 9);
322  pos = blockpos->byte_offset & 0x1FF;
323 
324  if(operation == 0)
325  { //If READ, copy latest data from the buffer of any FD that was opened for writing.
326  for(i = 0; i < pfsConfig.maxOpen; i++)
327  {
328  if((pfsFileSlots[i].clink != NULL)
329  && (pfsFileSlots[i].clink->pfsMount == pfsMount)
330  && (pfsFileSlots[i].unaligned.dirty)
331  && (pfsFileSlots[i].unaligned.sub == bi->subpart)
332  && (pfsFileSlots[i].unaligned.sector == sector))
333  {
334  unaligned2 = &pfsFileSlots[i].unaligned;
335  if(unaligned != unaligned2)
336  memcpy(unaligned->buffer, unaligned2->buffer, sizeof(unaligned->buffer));
337  else
338  unaligned2 = NULL; //Nothing to update.
339  break;
340  }
341  }
342  }
343 
344  if(unaligned2 == NULL)
345  { //If the latest version is likely in disk.
346  if((unaligned->sector != sector) || (unaligned->sub!=bi->subpart))
347  {
348  if (unaligned->dirty)
349  {
350  result=pfsMount->blockDev->transfer(pfsMount->fd, unaligned->buffer, unaligned->sub, unaligned->sector, 1, PFS_IO_MODE_WRITE);
351  if(result)
352  return result;
353  unaligned->dirty=0;
354  }
355 
356  unaligned->sub=bi->subpart;
357  unaligned->sector=sector;
358 
359  if(pos || (fileSlot->position != fileSlot->clink->u.inode->size))
360  {
361  result = pfsMount->blockDev->transfer(pfsMount->fd, unaligned->buffer, unaligned->sub, unaligned->sector, 1, PFS_IO_MODE_READ);
362  if(result)
363  return result | 0x10000;
364  }
365  }
366  }
367 
368  if (operation==0)
369  memcpy(buf, &unaligned->buffer[pos], size=min(size, (int)sizeof(unaligned->buffer)-(int)pos));
370  else
371  memcpy(&unaligned->buffer[pos], buf, size=min(size, (int)sizeof(unaligned->buffer)-(int)pos));
372 
373  if (operation == 1)
374  {
375  if (pfsMount->flags & PFS_FIO_ATTR_WRITEABLE)
376  {
377  if ((result=pfsMount->blockDev->transfer(pfsMount->fd, unaligned->buffer, unaligned->sub,
378  unaligned->sector, operation, operation)))
379  return result;
380 
381  //In late modules, sync data with all other FDs that were opened for reading.
382  for(i = 0; i < pfsConfig.maxOpen; i++)
383  {
384  if((pfsFileSlots[i].clink != NULL)
385  && (pfsFileSlots[i].clink->pfsMount == pfsMount)
386  && (pfsFileSlots[i].unaligned.sub == fileSlot->unaligned.sub)
387  && (pfsFileSlots[i].unaligned.sector == fileSlot->unaligned.sector))
388  {
389  if(!pfsFileSlots[i].unaligned.dirty)
390  memcpy(pfsFileSlots[i].unaligned.buffer, unaligned->buffer, sizeof(pfsFileSlots[i].unaligned.buffer));
391  }
392  }
393  }
394  else
395  unaligned->dirty=operation;
396  }
397  return size;
398 }

401 static int fileTransfer(pfs_file_slot_t *fileSlot, u8 *buf, int size, int operation)
402 {
403  int result=0;
404  pfs_blockpos_t *blockpos=&fileSlot->block_pos;
405  pfs_mount_t *pfsMount=fileSlot->clink->pfsMount;
406  u64 bytes_remain;
407  int total = size;
408 
409  // If we're writing and there is less free space in the last allocated block segment
410  // than can hold the data being written, then try and expand the block segment
411  if ((operation==1) &&
412  (fileSlot->clink->u.inode->number_data - 1 == blockpos->block_segment))
413  {
414  bytes_remain = (pfsBlockGetCurrent(blockpos)->count - blockpos->block_offset) * pfsMount->zsize - blockpos->byte_offset;
415 
416  if (bytes_remain < (u32)size)
417  {
418  pfsBlockExpandSegment(fileSlot->clink, blockpos,
419  ((size-bytes_remain+pfsMount->zsize-1) & (~(pfsMount->zsize-1))) / pfsMount->zsize);
420  }
421  }
422 
423  while (size)
424  {
425  pfs_blockinfo_t *bi;
426  u32 sectors;
427 
428  // Get block info for current file position
429  bi=pfsBlockGetCurrent(blockpos);
430 
431  // Get amount of remaining data in current block
432  bytes_remain=(bi->count - blockpos->block_offset) * pfsMount->zsize - blockpos->byte_offset;
433 
434  // If there is no space/data left in the current block segment, then we need to move onto the next
435  if (bytes_remain==0)
436  {
437  // If we're at the end of allocated block segments, then allocate a new block segment
438  if (blockpos->block_segment == fileSlot->clink->u.inode->number_data-1){
439  if (operation==0){
440  PFS_PRINTF(PFS_DRV_NAME" Panic: This is a bug!\n");
441  return 0;
442  }
443 
444  result = pfsBlockAllocNewSegment(fileSlot->clink, blockpos,
445  (size - 1 + pfsMount->zsize) / pfsMount->zsize);
446  if(result<0)
447  break;
448 
449  // Otherwise, move to the next block segment
450  }else
451  if ((result=pfsBlockSeekNextSegment(fileSlot->clink, blockpos)))
452  {
453  return result;
454  }
455 
456  bi=pfsBlockGetCurrent(blockpos);
457  bytes_remain=(bi->count - blockpos->block_offset) * pfsMount->zsize - blockpos->byte_offset;
458  }
459 
460  if (bytes_remain==0)
461  {
462  PFS_PRINTF(PFS_DRV_NAME" Panic: There is no zone.\n");
463  return 0;
464  }
465 
466  if(fileSlot->unaligned.dirty)
467  { //If there is going to be a partial transfer, write the dirty data to disk first.
468  u32 sector = ((bi->number + fileSlot->block_pos.block_offset) << pfsMount->sector_scale) + (fileSlot->block_pos.byte_offset / 512);
469  if((fileSlot->unaligned.sector == sector) &&
470  (fileSlot->unaligned.sub == bi->subpart)
471  && (size < 512))
472  {
473  if((result = pfsMount->blockDev->transfer(pfsMount->fd, fileSlot->unaligned.buffer, fileSlot->unaligned.sub, fileSlot->unaligned.sector, 1, PFS_IO_MODE_WRITE)) != 0)
474  return result;
475 
476  fileSlot->unaligned.dirty = 0;
477  }
478  }
479 
480  // If we are transferring size not a multiple of 512, under 512, or to
481  // an unaligned buffer we need to use a special function rather than
482  // just doing a ATA sector transfer.
483  if ((blockpos->byte_offset & 0x1FF) || (size < 512) || ((u32)buf & 3))
484  {
485  if ((result=fileTransferRemainder(fileSlot, buf, size, operation)) < 0)
486  break;
487  }
488  else
489  {
490  sectors = (u32)(bytes_remain / 512);
491  if ((u32)(size / 512) < sectors) sectors = size / 512; //sectors=min(size/512, sectors)
492 
493  // Do the ATA sector transfer
494  result=pfsMount->blockDev->transfer(pfsMount->fd, buf, bi->subpart,
495  ((bi->number + blockpos->block_offset) << pfsMount->sector_scale)+(blockpos->byte_offset / 512),
496  sectors, operation);
497  if (result < 0)
498  {
499  result |= 0x10000; // TODO: EIO define
500  break;
501  }
502  result = sectors * 512;
503  }
504 
505  size -= result;
506  fileSlot->position += result;
507  buf += result;
508 
509  // If file has grown, need to mark inode as dirty
510  if(fileSlot->clink->u.inode->size < fileSlot->position)
511  {
512  fileSlot->clink->u.inode->size = fileSlot->position;
513  fileSlot->clink->flags |= PFS_CACHE_FLAG_DIRTY;
514  }
515 
516  blockpos->block_offset+=pfsBlockSyncPos(blockpos, result);
517  }
518  return result < 0 ? result : total;
519 }
