96 int _start(int argc, char *argv[])
97 {

99  int threadid;
100 
101  FlushDcache();
102  CpuEnableIntr(0);
103 
104  if (argc>=1)
105  {
106  switch(argv[1][0])
107  {
108  case 'e': debug_level = OUT_ERROR; break; /* only errors */
109  case 'w': debug_level = OUT_WARNING; break; /* warnings too */
110  case 'i': debug_level = OUT_INFO; break; /* and info */
111  case 'd': debug_level = OUT_DEBUG; break; /* and debuging */
112  }
113  printf("debug_level = %d (%c)\n", debug_level, argv[1][0]);
114  }
115 
116  thread.attr = TH_C;
117  thread.thread = rpc_thread;
118  thread.priority = 40;
119  thread.stacksize = 0x800;
120  thread.option = 0;
121  threadid = CreateThread(&thread);
122  if (threadid < 0)
123  {
124  dprintf(OUT_ERROR, "Failed to make thread\n");
125  return(1);
126  }
127 
128  StartThread(threadid, 0);
129 
130  return(0);
