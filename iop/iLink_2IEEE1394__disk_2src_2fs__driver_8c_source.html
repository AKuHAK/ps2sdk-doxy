216 static int fs_open(iop_file_t* fd, const char *name, int flags, int mode) {
217  fat_driver* fatd;
218  fs_rec* rec, *rec2;
219  int ret;
220  unsigned int cluster;
221  char escapeNotExist;
222 
223  _fs_lock();
224 
225  XPRINTF("USBHDFSD: fs_open called: %s flags=%X mode=%X \n", name, flags, mode) ;
226 
227  fatd = fat_getData(fd->unit);
228  if (fatd == NULL) { _fs_unlock(); return -ENODEV; }
229 
230  //check if the slot is free
231  rec = fs_findFreeFileSlot();
232  if (rec == NULL) { _fs_unlock(); return -EMFILE; }
233 
234  //find the file
235  cluster = 0; //allways start from root
236  XPRINTF("USBHDFSD: Calling fat_getFileStartCluster from fs_open\n");
237  ret = fat_getFileStartCluster(fatd, name, &cluster, &rec->dirent.fatdir);
238  if (ret < 0 && ret != -ENOENT) {
239  _fs_unlock();
240  return ret;
241  }else{
242  //File exists. Check if the file is already open
243  rec2 = fs_findFileSlotByCluster(rec->dirent.fatdir.startCluster);
244  if (rec2 != NULL) {
245  if ((flags & O_WRONLY) || //current file is opened for write
246  (rec2->mode & O_WRONLY) ) {//other file is opened for write
247  _fs_unlock();
248  return -EACCES;
249  }
250  }
251  }
252 
253  if(flags & O_WRONLY) { //dlanor: corrected bad test condition
254  cluster = 0; //start from root
255 
256  escapeNotExist = 1;
257  if (flags & O_CREAT) {
258  XPRINTF("USBHDFSD: FAT I: O_CREAT detected!\n");
259  escapeNotExist = 0;
260  }
261 
262  rec->sfnSector = 0;
263  rec->sfnOffset = 0;
264  ret = fat_createFile(fatd, name, 0, escapeNotExist, &cluster, &rec->sfnSector, &rec->sfnOffset);
265  if (ret < 0) {
266  _fs_unlock();
267  return ret;
268  }
269  //the file already exist but flags is set to truncate
270  if (ret == 2 && (flags & O_TRUNC)) {
271  XPRINTF("USBHDFSD: FAT I: O_TRUNC detected!\n");
272  fat_truncateFile(fatd, cluster, rec->sfnSector, rec->sfnOffset);
273  }
274 
275  //find the file
276  cluster = 0; //allways start from root
277  XPRINTF("USBHDFSD: Calling fat_getFileStartCluster from fs_open after file creation\n");
278  ret = fat_getFileStartCluster(fatd, name, &cluster, &rec->dirent.fatdir);
279  }
280 
281  if (ret < 0) { //At this point, the file should be locatable without any errors.
282  _fs_unlock();
283  return ret;
284  }
285 
286  if ((rec->dirent.fatdir.attr & FAT_ATTR_DIRECTORY) == FAT_ATTR_DIRECTORY) {
287  // Can't open a directory with open
288  _fs_unlock();
289  return -EISDIR;
290  }
291 
292  rec->dirent.file_flag = FS_FILE_FLAG_FILE;
293  rec->mode = flags;
294  rec->filePos = 0;
295  rec->sizeChange = 0;
296 
297  if ((flags & O_APPEND) && (flags & O_WRONLY)) {
298  XPRINTF("USBHDFSD: FAT I: O_APPEND detected!\n");
299  rec->filePos = rec->dirent.fatdir.size;
300  }
301 
302  //store the slot to user parameters
303  fd->privdata = rec;
304 
305  _fs_unlock();
306  return 1;
307 }

344 static int fs_lseek(iop_file_t* fd, int offset, int whence) {
345  fat_driver* fatd;
346  fs_rec* rec = (fs_rec*)fd->privdata;
347 
348  if (rec == NULL)
349  return -EBADF;
350 
351  _fs_lock();
352 
353  fatd = fat_getData(fd->unit);
354  if (fatd == NULL) { _fs_unlock(); return -ENODEV; }
355 
356  if (rec->dirent.file_flag != FS_FILE_FLAG_FILE) {
357  _fs_unlock();
358  return -EISDIR;
359  }
360 
361  switch(whence) {
362  case SEEK_SET:
363  rec->filePos = offset;
364  break;
365  case SEEK_CUR:
366  rec->filePos += offset;
367  break;
368  case SEEK_END:
369  rec->filePos = rec->dirent.fatdir.size + offset;
370  break;
371  default:
372  _fs_unlock();
373  return -1;
374  }
375  if (rec->filePos < 0) {
376  rec->filePos = 0;
377  }
378  if (rec->filePos > rec->dirent.fatdir.size) {
379  rec->filePos = rec->dirent.fatdir.size;
380  }
381 
382  _fs_unlock();
383  return rec->filePos;
384 }

387 static int fs_write(iop_file_t* fd, void * buffer, int size )
388 {
389  fat_driver* fatd;
390  fs_rec* rec = (fs_rec*)fd->privdata;
391  int result;
392  int updateClusterIndices = 0;
393 
394  if (rec == NULL)
395  return -EBADF;
396 
397  _fs_lock();
398 
399  fatd = fat_getData(fd->unit);
400  if (fatd == NULL) { _fs_unlock(); return -ENODEV; }
401 
402  if (rec->dirent.file_flag != FS_FILE_FLAG_FILE) {
403  _fs_unlock();
404  return -EISDIR;
405  }
406 
407  if (!(rec->mode & O_WRONLY)) {
408  _fs_unlock();
409  return -EACCES;
410  }
411 
412  if (size <= 0) { _fs_unlock(); return 0; }
413 
414  result = fat_writeFile(fatd, &rec->dirent.fatdir, &updateClusterIndices, rec->filePos, buffer, size);
415  if (result > 0) { //write succesful
416  rec->filePos += result;
417  if (rec->filePos > rec->dirent.fatdir.size) {
418  rec->dirent.fatdir.size = rec->filePos;
419  rec->sizeChange = 1;
420  //if new clusters allocated - then update file cluster indices
421  if (updateClusterIndices) {
422  fat_setFatDirChain(fatd, &rec->dirent.fatdir);
423  }
424  }
425  }
426 
427  _fs_unlock();
428  return result;
429 }
