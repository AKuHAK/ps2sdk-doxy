43 unsigned short int iLinkCalculateCRC16(void *data, unsigned int nQuads){
44  int i, lShift;
45  unsigned int lData, lSum;
46  unsigned short int lCRC = 0;
47 
48  for ( i = 0; i < nQuads; ++i ) {
49  lData = ((unsigned int *)data)[i];
50  for ( lShift = 28; lShift >= 0; lShift -= 4 ) {
51  lSum = ( ( lCRC >> 12 ) ^ ( lData >> lShift ) ) & 15;
52  lCRC = ( lCRC << 4 ) ^ ( lSum << 12 ) ^ ( lSum << 5 ) ^ lSum;
53  } /* end for */
54  lCRC &= 0xFFFF;
55  } /* end for */
56 
57  return lCRC;
