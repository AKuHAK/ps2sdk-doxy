226 u32 padTransfer(u32 slot)
227 {
228  u32 stat70;
229  u32 trans_count = 0, port0_start = 0, port1_start = 0;
230  u32 val_port0 = 0;
231  u32 val_port1 = 0;
232 
233  sio2_td.in_size = 0;
234  sio2_td.out_size = 0;
235  sio2_td.in_dma.addr = 0;
236  sio2_td.out_dma.addr = 0;
237 
238  stat70 = sio2_stat70_get();
239 
240  pdSetStat70bit(0, slot, (stat70 >> 4) & 1);
241  pdSetStat70bit(1, slot, (stat70 >> 5) & 1);
242 
243  if( padData[0][slot].active == 1 )
244  {
245  if(change_slot_buffer[4] == 1)
246  {
247  val_port0 = 1;
248  port0_start = 0;
249  trans_count = setupTransferData(0, 0, slot);
250  }
251  }
252 
253  if( padData[1][slot].active == 1 )
254  {
255  if(change_slot_buffer[5] == 1)
256  {
257  val_port1 = 1;
258  port1_start = trans_count;
259  trans_count = setupTransferData(trans_count, 1, slot);
260  }
261  }
262 
263  /* These two blocks of code change position before v3.6.
264  At v3.3 and earlier, they were within the transfer block below. At v3.0, there were no checks on change_slot_buffer. */
265  if(padData[0][slot].active == 1)
266  {
267  if(change_slot_buffer[4] != 1)
268  padData[0][slot].error = 0xA;
269  }
270 
271  if(padData[1][slot].active == 1)
272  {
273  if(change_slot_buffer[5] != 1)
274  padData[1][slot].error = 0xA;
275  }
276 
277  if(trans_count != 0)
278  {
279  sio2_transfer2( &sio2_td );
280 
281  sio2_td.out_size = 0;
282 
283  if(val_port0 == 1)
284  {
285  readSio2OutBuffer(port0_start, 0, slot);
286  sio2_td.out_size = padData[0][slot].out_size;
287  }
288 
289  if(val_port1 == 1)
290  {
291  readSio2OutBuffer(port1_start, 1, slot);
292  sio2_td.out_size = padData[1][slot].out_size;
293  }
294 
295  return 1;
296  }
297 
298  return 0;

383  sio2_td.out_size = padData[port][slot].out_size;

385  //This used to be per port too.
