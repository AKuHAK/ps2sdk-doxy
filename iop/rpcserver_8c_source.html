173 static void* RpcServer(int fno, void *buffer, int length)
174 {
175  u32 *data = (u32*)buffer;
176 
177  switch(data[0])
178  {
179  case PAD_RPCCMD_INIT: return RpcPadInit(data);
180  case PAD_RPCCMD_END: return RpcPadEnd(data);
181  case PAD_RPCCMD_GET_MODVER: return RpcGetModVersion(data);
182  case PAD_RPCCMD_OPEN: return RpcPadOpen(data);
183  case PAD_RPCCMD_CLOSE: return RpcPadClose(data);
184  case PAD_RPCCMD_INFO_ACT: return RpcPadInfoAct(data);
185  case PAD_RPCCMD_INFO_COMB: return RpcPadInfoComb(data);
186  case PAD_RPCCMD_INFO_MODE: return RpcPadInfoMode(data);
187  case PAD_RPCCMD_SET_MMODE: return RpcPadSetMainMode(data);
188  case PAD_RPCCMD_SET_ACTDIR: return RpcPadSetActDirect(data);
189  case PAD_RPCCMD_SET_ACTALIGN: return RpcPadSetActAlign(data);
190  case PAD_RPCCMD_GET_BTNMASK: return RpcPadGetButtonMask(data);
191  case PAD_RPCCMD_SET_BTNINFO: return RpcPadSetButtonInfo(data);
192  case PAD_RPCCMD_SET_VREF: return RpcPadSetVrefParam(data);
193  case PAD_RPCCMD_GET_PORTMAX: return RpcPadGetPortMax(data);
194  case PAD_RPCCMD_GET_SLOTMAX: return RpcPadGetSlotMax(data);
195 
196  default:
197  M_PRINTF("invalid function code (%03x)\n", (int)data[0]);
198  break;
199  }
200 
201  return buffer;
202 }
