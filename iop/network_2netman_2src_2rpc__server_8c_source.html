167 
168  if(TxFrameBuffer == NULL) TxFrameBuffer = malloc(NETMAN_MAX_FRAME_SIZE * NETMAN_RPC_BLOCK_SIZE);
169  if(FrameBufferStatus == NULL) FrameBufferStatus = malloc(sizeof(struct NetManBD) * NETMAN_RPC_BLOCK_SIZE);
170 
171  if(TxFrameBuffer != NULL && FrameBufferStatus != NULL)
172  {
173  EEFrameBufferStatus = ((struct NetManRegNetworkStack*)buffer)->FrameBufferStatus;
174  memset(FrameBufferStatus, 0, sizeof(struct NetManBD) * NETMAN_RPC_BLOCK_SIZE);

177  IOPFrameBufferRdPtr = 0;
178  ResultValue = NetManRegisterNetworkStack(&RpcStack);
179 
180  if(ResultValue == 0)
