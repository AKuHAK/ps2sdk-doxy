178 void main_thread(void *unused)
179 {
180  u32 resbits[4];
181 
182  while (1) {
183  #ifdef SIO2LOG
184  log_flush(0);
185  #endif
186  WaitEventFlag(event_flag, EF_PAD_TRANSFER_INIT |
187  EF_MC_TRANSFER_INIT | EF_MTAP_TRANSFER_INIT
188 #ifdef SIO2MAN_V2
189  | EF_RM_TRANSFER_INIT | EF_UNK_TRANSFER_INIT
190 #endif
191  , 1, resbits);
192 
193  if (resbits[0] & EF_PAD_TRANSFER_INIT) {
194  ClearEventFlag(event_flag, ~EF_PAD_TRANSFER_INIT);
195  SetEventFlag(event_flag, EF_PAD_TRANSFER_READY);
196 #ifdef SIO2LOG
197  log_default(LOG_PAD_READY);
198 #endif
199  } else if (resbits[0] & EF_MC_TRANSFER_INIT) {
200  ClearEventFlag(event_flag, ~EF_MC_TRANSFER_INIT);
201  SetEventFlag(event_flag, EF_MC_TRANSFER_READY);
202 #ifdef SIO2LOG
203  log_default(LOG_MC_READY);
204 #endif
205  } else if (resbits[0] & EF_MTAP_TRANSFER_INIT) {
206  ClearEventFlag(event_flag, ~EF_MTAP_TRANSFER_INIT);
207  SetEventFlag(event_flag, EF_MTAP_TRANSFER_READY);
208 #ifdef SIO2LOG
209  log_default(LOG_MTAP_READY);
210 #endif
211 #ifdef SIO2MAN_V2
212  } else if (resbits[0] & EF_RM_TRANSFER_INIT) {
213  ClearEventFlag(event_flag, ~EF_RM_TRANSFER_INIT);
214  SetEventFlag(event_flag, EF_RM_TRANSFER_READY);
215 #ifdef SIO2LOG
216  log_default(LOG_RM_READY);
217 #endif
218  } else if (resbits[0] & EF_UNK_TRANSFER_INIT) {
219  ClearEventFlag(event_flag, ~EF_UNK_TRANSFER_INIT);
220  SetEventFlag(event_flag, EF_UNK_TRANSFER_READY);
221 #ifdef SIO2LOG
222  log_default(LOG_UNK_READY);
223 #endif
224 #endif
225  } else {
226  EPRINTF("Unknown event %08lx. Exiting.\n", resbits[0]);
227  return;
228  }
229 
230 transfer_loop:
231  WaitEventFlag(event_flag, EF_TRANSFER_START | EF_TRANSFER_RESET, 1, resbits);
232 
233  if (resbits[0] & EF_TRANSFER_RESET) {
234  ClearEventFlag(event_flag, ~EF_TRANSFER_RESET);
235 #ifdef SIO2LOG
236  log_default(LOG_RESET);
237 #endif
238  continue;
239  }
240 
241  ClearEventFlag(event_flag, ~EF_TRANSFER_START);
242 
243  sio2_ctrl_set(sio2_ctrl_get() | 0xc);
244  send_td(transfer_data);
245  sio2_ctrl_set(sio2_ctrl_get() | 1);
246 
247  WaitEventFlag(event_flag, EF_SIO2_INTR_COMPLETE, 0, NULL);
248  ClearEventFlag(event_flag, ~EF_SIO2_INTR_COMPLETE);
249 
250  recv_td(transfer_data);
251  SetEventFlag(event_flag, EF_TRANSFER_FINISH);
252 
253  /* Bah... this is needed to get the initial dump from XMCMAN,
254  but it will kill the IOP when XPADMAN is spamming...
255 
256  TODO
257  I guess the correct solution is to do all logging in a
258  dedicated thread. */
259 // log_flush(1);
260 
261  goto transfer_loop;
262  }
263 }

302 int _start(int argc, const char **argv)
303 {
304  int state;
305 
306  shutdown();
307 
308  if (RegisterLibraryEntries(&_exp_sio2man) != 0)
309  return 1;
310 
311  if (init)
312  return 1;
313 
314  init = 1;
315 
316  sio2_ctrl_set(0x3bc);
317 
318  mtap_change_slot_cb = NULL; mtap_get_slot_max_cb = NULL; mtap_get_slot_max2_cb = NULL; mtap_update_slots_cb = NULL;
319  event_flag = create_event_flag();
320  thid = create_main_thread();
321 
322  CpuSuspendIntr(&state);
323  RegisterIntrHandler(IOP_IRQ_SIO2, 1, sio2_intr_handler, &event_flag);
324  EnableIntr(IOP_IRQ_SIO2);
325  CpuResumeIntr(state);
326 
327  dmac_ch_set_dpcr(IOP_DMAC_SIO2in, 3);
328  dmac_ch_set_dpcr(IOP_DMAC_SIO2out, 3);
329  dmac_enable(IOP_DMAC_SIO2in);
330  dmac_enable(IOP_DMAC_SIO2out);
331 
332  StartThread(thid, NULL);
333 #ifdef SIO2LOG
334  EPRINTF("Logging started.\n");
335 #endif
336  return 0;
