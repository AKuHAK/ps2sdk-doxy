66 int TransInterrupt(void *data)
67 {
68  IntrData *intr = (IntrData*) data;
69  s32 dir;
70  s32 core;
71 
72  dir = ((intr->mode & 0x200) < 1);
73  core = intr->mode & 0xFF;
74 
75  // Voice Transfer
76  if((intr->mode & 0x100) == 0)
77  {
78  #ifndef ISJPCM
79  // SD_C_STATX(core)
80  // If done elsewise, it doesn't work, havn't figured out why yet.
81  volatile u16 *statx = U16_REGISTER(0x344 + (core * 1024));
82 
83  if(!(*statx & 0x80)) while(!(*statx & 0x80));
84 
85  *SD_CORE_ATTR(core) &= ~SD_CORE_DMA;
86 
87  if(*SD_CORE_ATTR(core) & 0x30) while((*SD_CORE_ATTR(core) & 0x30));
88 
89  if(TransIntrHandlers[core]) goto intr_handler;
90  if(TransIntrCallbacks[core]) goto SdIntrCallback;
91 
92  VoiceTransComplete[core] = 1;
93  #endif
94  }
95  else
96  { // Block Transfer
97  if(intr->mode & (SD_TRANS_LOOP << 8))
98  {
99  // Switch buffers
100  BlockTransBuff[core] = 1 - BlockTransBuff[core];
101  // Setup DMA & send
102  *SD_DMA_ADDR(core) = (BlockTransSize[core] * BlockTransBuff[core])+BlockTransAddr[core];
103  *SD_DMA_SIZE(core) = (BlockTransSize[core]/64)+((BlockTransSize[core]&63)>0);
104  *SD_DMA_CHCR(core) = SD_DMA_START | SD_DMA_CS | dir;
105  }
106  else
107  {
108  #ifndef ISJPCM
109  *SD_CORE_ATTR(core) &= ~SD_CORE_DMA;
110  *SD_P_MMIX(core) &= 0xFF3F;
111  *U16_REGISTER(0x1B0 + (core * 1024)) = 0;
112  #endif
113  }
114 
115  if(TransIntrHandlers[core])
116  {
117  #ifndef ISJPCM
118  intr_handler:
119  #endif
120  TransIntrHandlers[core](core, intr->data);
121  }
122  else
123  {
124  if(TransIntrCallbacks[core])
125  {
126  #ifndef ISJPCM
127  SdIntrCallback:
128  #endif
129  TransIntrCallbacks[core](0);
130  }
131  }
132  }
133 
134  if(dir == SD_DMA_DIR_SPU2IOP) FlushDcache();
135 
136  return 1;
137 }

498 void sceSdSetCoreAttr(u16 entry, u16 val)
499 {
500  u16 core_attr = *SD_CORE_ATTR(entry & 1);
501 
502  switch(entry & ~1)
503  {
504  case SD_CORE_NOISE_CLK: // 0x8
505  *SD_CORE_ATTR(entry & 1) = (core_attr-0x3F01) | ((val & 0x3F) << 8);
506  break;
507 
508  #ifndef ISJPCM
509  case SD_CORE_SPDIF_MODE: // 0xA
510  SetSpdifMode(val); // sub1
511  break;
512  #endif
513  default:
514  {
515  u32 core = entry & 1;
516  entry = (entry >> 1)-1;
517  core_attr &= ~(1 << CoreAttrShifts[entry]);
518  core_attr |= (val & 1) << CoreAttrShifts[entry];
519  *SD_CORE_ATTR(core) = core_attr;
520  }
521  break;
522  }
523 }

622 s32 BlockTransWriteFrom(u8 *iopaddr, u32 size, s32 chan, u16 mode, u8 *startaddr)
623 {
624  s32 core = chan & 1;
625  s32 offset;
626 
627  BlockTransBuff[core] = 0;
628  BlockTransSize[core] = size;
629  BlockTransAddr[core] = (u32)iopaddr;
630 
631  if(startaddr == 0)
632  startaddr = iopaddr;
633 
634  offset = startaddr - iopaddr;
635 
636  if(offset > size)
637  {
638  if(mode & SD_TRANS_LOOP)
639  {
640  offset -= size;
641  BlockTransBuff[core] = 1;
642  }
643  else
644  {
645  return -1;
646  }
647  }
648 
649  if(offset & 1023) offset += 1024;
650 
651  iopaddr += (BlockTransSize[core] * BlockTransBuff[core]) + offset;
652 
653  if(*SD_CORE_ATTR(core) & SD_DMA_IN_PROCESS) return -1;
654  if(*SD_DMA_CHCR(core) & SD_DMA_START) return -1;

658 
659  *SD_A_TSA_HI(core) = 0;
660  *SD_A_TSA_LO(core) = 0;

662  *U16_REGISTER(0x1B0+(core*1024)) = 1 << core;
663

665

667  *SD_DMA_MODE(core) = 0x10;
668  *SD_DMA_SIZE(core) = (size/64)+((size&63)>0);
