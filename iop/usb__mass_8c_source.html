293 static int usb_bulk_manage_status(mass_dev* dev, unsigned int tag) {
294  int ret;
295  csw_packet csw;
296 
297  //XPRINTF("USBHDFSD: usb_bulk_manage_status 1 ...\n");
298  ret = usb_bulk_status(dev, &csw, tag); /* Attempt to read CSW from bulk in endpoint */
299  if (ret != USB_RC_OK) { /* STALL bulk in -OR- Bulk error */
300  usb_bulk_clear_halt(dev, USB_BLK_EP_IN); /* clear the stall condition for bulk in */

304  }

308  M_DEBUG("ERROR: usb_bulk_manage_status call reset recovery ...\n");
309  usb_bulk_reset(dev, 3); /* Perform reset recovery */
310  }
311 
312  return((ret == USB_RC_OK && csw.signature == CSW_TAG && csw.tag == tag) ? csw.status : -1);

437 int usb_queue_cmd(struct scsi_interface* scsi, const unsigned char* cmd, unsigned int cmd_len, unsigned char* data, unsigned int data_len, unsigned int data_wr)
438 {
439  mass_dev* dev = (mass_dev*)scsi->priv;
440  static struct usbmass_cmd ucmd;
441  int result;
442 #ifndef ASYNC
443  int rcode;
444 #endif
445  static unsigned int tag = 0;
446 
447  M_DEBUG("%s\n", __func__);
448 
449  tag++;
450 
451  // Create USB command
452  ucmd.dev = dev;
453  ucmd.cmd_count = 0;
454 
455  // Create CBW
456  ucmd.cbw.signature = CBW_TAG;
457  ucmd.cbw.tag = tag;
458  ucmd.cbw.dataTransferLength = data_len;
459  ucmd.cbw.flags = data_wr ? 0 : 0x80;
460  ucmd.cbw.lun = 0;
461  ucmd.cbw.comLength = cmd_len;
462  memcpy(ucmd.cbw.comData, cmd, cmd_len);
463 
464  // Create CSW
465  ucmd.csw.signature = CSW_TAG;
466  ucmd.csw.tag = tag;
467  ucmd.csw.dataResidue = 0;
468  ucmd.csw.status = 0;
469 
470 #ifndef ASYNC
471  result = -EIO;
472  if(usb_bulk_command(dev, &ucmd.cbw) == USB_RC_OK){
473  if (data_len > 0)
474  rcode = usb_bulk_transfer(dev, data_wr ? USB_BLK_EP_OUT : USB_BLK_EP_IN, data, data_len);
475  else
476  rcode = USB_RC_OK;
477 
478  result = usb_bulk_manage_status(dev, tag);
479 
480  if(rcode != USB_RC_OK)
481  result = -EIO;
482  }
483 
484  return result;
485 #else
486  // Send the CBW (command)
487  ucmd.cmd_count++;
488  result = sceUsbdBulkTransfer(dev->bulkEpO, &ucmd.cbw, 31, scsi_cmd_callback, (void*)&ucmd);
489  if (result != USB_RC_OK)
490  return -EIO;
491 
492  // Send/Receive data
493  while (data_len > 0) {
494  unsigned int tr_len = (data_len < USB_BLOCK_SIZE) ? data_len : USB_BLOCK_SIZE;
495  ucmd.cmd_count++;
496  result = sceUsbdBulkTransfer(data_wr ? dev->bulkEpO : dev->bulkEpI, data, tr_len, scsi_cmd_callback, (void*)&ucmd);
497  if (result != USB_RC_OK)
498  return -EIO;
499  data_len -= tr_len;
500  data += tr_len;
501  }
502 
503  // Receive CSW (status)
504  ucmd.cmd_count++;
505  result = sceUsbdBulkTransfer(dev->bulkEpI, &ucmd.csw, 13, scsi_cmd_callback, (void*)&ucmd);
506  if (result != USB_RC_OK)
507  return -EIO;
508 
509  // Wait for SCSI command to finish
510  WaitSema(dev->ioSema);
511  if (ucmd.returnCode != USB_RC_OK)
512  return -EIO;
513 
514  return 0;
515 #endif
516 }

590  M_DEBUG("ERROR: No interfaces available\n");

599  return 0; //we send the CBW to te bulk out endpoint
600  }

723 static void usb_mass_update(void* arg)
724 {
725  int i;
726 
727  M_DEBUG("update thread running\n");
728 
729  while (1) {
730  // Wait for event from USBD thread
731  WaitSema(usb_mass_update_sema);
732 
733  // Connect new devices
734  for (i = 0; i < NUM_DEVICES; ++i) {
735  mass_dev* dev = &g_mass_device[i];
736  if (dev->devId != -1 && (dev->status & USBMASS_DEV_STAT_CONN) && !(dev->status & USBMASS_DEV_STAT_CONF)) {
737  int ret;
738  if ((ret = usb_set_configuration(dev, dev->configId)) != USB_RC_OK) {
739  M_PRINTF("ERROR: sending set_configuration %d\n", ret);
740  usb_mass_release(dev);
741  continue;
742  }
743 
744  if ((ret = usb_set_interface(dev, dev->interfaceNumber, dev->interfaceAlt)) != USB_RC_OK) {
745  M_PRINTF("ERROR: sending set_interface %d\n", ret);
746  if (ret == USB_RC_STALL) {
747  /* USB Specification 1.1, section 9.4.10: Devices that only support a default setting for the specified interface may return a STALL.
748  As with Linux, we shall clear the halt state of the interface's pipes and continue. */
749  usb_bulk_clear_halt(dev, USB_BLK_EP_IN);
750  usb_bulk_clear_halt(dev, USB_BLK_EP_OUT);
751  } else {
752  usb_mass_release(dev);
753  continue;
754  }
755  }
756 
757  dev->status |= USBMASS_DEV_STAT_CONF;
758  scsi_connect(&dev->scsi);
759  }
760  }
761  }
