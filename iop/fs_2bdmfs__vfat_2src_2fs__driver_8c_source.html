222 static int fs_open(iop_file_t* fd, const char* name, int flags, int mode)
223 {
224  fat_driver* fatd;
225  fs_rec *rec, *rec2;
226  int ret;
227  unsigned int cluster;
228  char escapeNotExist;
229 
230  M_DEBUG("%s: %s flags=%X mode=%X\n", __func__, name, flags, mode);
231 
232  _fs_lock();
233 
234  fatd = fat_getData(fd->unit);
235  if (fatd == NULL) {
236  _fs_unlock();
237  return -ENODEV;
238  }
239 
240  //check if the slot is free
241  rec = fs_findFreeFileSlot();
242  if (rec == NULL) {
243  _fs_unlock();
244  return -EMFILE;
245  }
246 
247  //find the file
248  cluster = 0; //allways start from root
249  M_DEBUG("Calling fat_getFileStartCluster from fs_open\n");
250  ret = fat_getFileStartCluster(fatd, name, &cluster, &rec->dirent.fatdir);
251  if (ret < 0 && ret != -ENOENT) {
252  _fs_unlock();
253  return ret;
254  } else {
255  //File exists. Check if the file is already open
256  rec2 = fs_findFileSlotByCluster(rec->dirent.fatdir.startCluster);
257  if (rec2 != NULL) {
258  if ((flags & O_WRONLY) || //current file is opened for write
259  (rec2->mode & O_WRONLY)) { //other file is opened for write
260  _fs_unlock();
261  return -EACCES;
262  }
263  }
264  }
265 
266  if (flags & O_WRONLY) { //dlanor: corrected bad test condition
267  cluster = 0; //start from root
268 
269  escapeNotExist = 1;
270  if (flags & O_CREAT) {
271  M_DEBUG("FAT I: O_CREAT detected!\n");
272  escapeNotExist = 0;
273  }
274 
275  rec->sfnSector = 0;
276  rec->sfnOffset = 0;
277  ret = fat_createFile(fatd, name, 0, escapeNotExist, &cluster, &rec->sfnSector, &rec->sfnOffset);
278  if (ret < 0) {
279  FLUSH_SECTORS(fatd);
280  _fs_unlock();
281  return ret;
282  }
283  //the file already exist but flags is set to truncate
284  if (ret == EEXIST && (flags & O_TRUNC)) {
285  M_DEBUG("FAT I: O_TRUNC detected!\n");
286  ret = fat_truncateFile(fatd, cluster, rec->sfnSector, rec->sfnOffset);
287  if (ret < 0) {
288  FLUSH_SECTORS(fatd);
289  M_DEBUG("FAT E: failed to truncate!\n");
290  _fs_unlock();
291  return ret;
292  }
293  }
294 
295  //find the file
296  cluster = 0; //allways start from root
297  M_DEBUG("Calling fat_getFileStartCluster from fs_open after file creation\n");
298  ret = fat_getFileStartCluster(fatd, name, &cluster, &rec->dirent.fatdir);
299  }
300 
301  if (ret < 0) { //At this point, the file should be locatable without any errors.
302  _fs_unlock();
303  return ret;
304  }
305 
306  if ((rec->dirent.fatdir.attr & FAT_ATTR_DIRECTORY) == FAT_ATTR_DIRECTORY) {
307  // Can't open a directory with open
308  _fs_unlock();
309  return -EISDIR;
310  }
311 
312  rec->dirent.file_flag = FS_FILE_FLAG_FILE;
313  rec->mode = flags;
314  rec->filePos = 0;
315  rec->sizeChange = 0;
316 
317  if ((flags & O_APPEND) && (flags & O_WRONLY)) {
318  M_DEBUG("FAT I: O_APPEND detected!\n");
319  rec->filePos = rec->dirent.fatdir.size;
320  }
321 
322  //store the slot to user parameters
323  fd->privdata = rec;
324 
325  _fs_unlock();
326  return 1;
327 }

370 static int fs_lseek(iop_file_t* fd, int offset, int whence)
371 {
372  fat_driver* fatd;
373  fs_rec* rec = (fs_rec*)fd->privdata;
374 
375  M_DEBUG("%s\n", __func__);
376 
377  if (rec == NULL)
378  return -EBADF;
379 
380  _fs_lock();
381 
382  fatd = fat_getData(fd->unit);
383  if (fatd == NULL) {
384  _fs_unlock();
385  return -ENODEV;
386  }
387 
388  if (rec->dirent.file_flag != FS_FILE_FLAG_FILE) {
389  _fs_unlock();
390  return -EISDIR;
391  }
392 
393  switch (whence) {
394  case SEEK_SET:
395  rec->filePos = offset;
396  break;
397  case SEEK_CUR:
398  rec->filePos += offset;
399  break;
400  case SEEK_END:
401  rec->filePos = rec->dirent.fatdir.size + offset;
402  break;
403  default:
404  _fs_unlock();
405  return -1;
406  }
407  if (rec->filePos < 0) {
408  rec->filePos = 0;
409  }
410  if (rec->filePos > rec->dirent.fatdir.size) {
411  rec->filePos = rec->dirent.fatdir.size;
412  }
413 
414  _fs_unlock();
415  return rec->filePos;
416 }
