133 apa_cache_t *apaInsertPartition(s32 device, const apa_params_t *params, u32 sector, int *err)
134 { // Adds a new partition using an empty block.
135  apa_cache_t *clink_empty;
136  apa_cache_t *clink_this;
137  apa_cache_t *clink_next;
138 
139  if((clink_this=apaCacheGetHeader(device, sector, APA_IO_MODE_READ, err))==0)
140  return 0;
141 
142  while(clink_this->header->length!=params->size)
143  {
144  if((clink_next=apaCacheGetHeader(device, clink_this->header->next, APA_IO_MODE_READ, err))==NULL)
145  { // Get next partition
146  apaCacheFree(clink_this);
147  return 0;
148  }
149  clink_this->header->length>>=1;
150  clink_empty=apaRemovePartition(device, (clink_this->header->start+clink_this->header->length),
151  clink_this->header->next, clink_this->header->start, clink_this->header->length);
152  clink_this->header->next=clink_empty->header->start;
153  clink_this->flags|=APA_CACHE_FLAG_DIRTY;
154  clink_next->header->prev=clink_empty->header->start;
155  clink_next->flags|=APA_CACHE_FLAG_DIRTY;
156 
157  apaCacheFlushAllDirty(device);
158  apaCacheFree(clink_empty);
159  apaCacheFree(clink_next);
160  }
161  apaCacheFree(clink_this);
162  clink_this=apaFillHeader(device, params, clink_this->header->start, clink_this->header->next,
163  clink_this->header->prev, params->size, err);
164  apaCacheFlushAllDirty(device);
165  return clink_this;
166 }

249 apa_cache_t *apaDeleteFixPrev(apa_cache_t *clink, int *err)
250 {
251  apa_cache_t *clink2=clink;
252  apa_header_t *header=clink2->header;
253  u32 device=clink->device;
254  u32 length=clink->header->length;
255  u32 saved_next=clink->header->next;
256  u32 saved_length=clink->header->length;
257  u32 tmp;
258 
259  while(header->start)
260  {
261  if(!(clink2=apaCacheGetHeader(device, header->prev, APA_IO_MODE_READ, err)))
262  {
263  apaCacheFree(clink);
264  return NULL;
265  }
266  header=clink2->header;
267  tmp=header->length+length;
268  if(header->type!=0) {
269  apaCacheFree(clink2);
270  break;
271  }
272  if((header->start%tmp) || (tmp & (tmp-1))) {
273  apaCacheFree(clink2);
274  break;
275  }
276  length=tmp;
277  apaCacheFree(clink);
278  clink=clink2;
279  }
280  if(length!=saved_length)
281  {
282  if(!(clink2=apaCacheGetHeader(device, saved_next, APA_IO_MODE_READ, err)))
283  {
284  apaCacheFree(clink);
285  return NULL;
286  }
287  clink->header->length=length;
288  clink->header->next=clink->header->start+length;
289  clink2->header->prev=clink->header->start;
290  clink2->flags|=APA_CACHE_FLAG_DIRTY;
291  clink->flags|=APA_CACHE_FLAG_DIRTY;
292  apaCacheFlushAllDirty(device);
293  apaCacheFree(clink2);
294  }
295  return clink;
296 }

352 int apaDelete(apa_cache_t *clink)
353 {
354  int rv=0;
355  apa_cache_t *clink_mbr;
356  u32 device=clink->device;
357  u32 start=clink->header->start;
358  int i;
359 
360  if(!start) {
361  apaCacheFree(clink);
362  return -EACCES;
363  }
364 
365  if(clink->header->next==0)
366  {
367  if((clink_mbr=apaCacheGetHeader(device, 0, APA_IO_MODE_READ, &rv))==NULL)
368  {
369  apaCacheFree(clink);
370  return rv;
371  }
372  do {
373  apaCacheFree(clink);
374  if((clink=apaCacheGetHeader(clink->device, clink->header->prev, APA_IO_MODE_READ, &rv))==NULL)
375  return 0;
376  clink->header->next=0;
377  clink->flags|=APA_CACHE_FLAG_DIRTY;
378  clink_mbr->header->prev=clink->header->start;
379  clink_mbr->flags|=APA_CACHE_FLAG_DIRTY;
380  apaCacheFlushAllDirty(device);
381  } while(clink->header->type==0);
382  apaCacheFree(clink_mbr);
383  } else {
384  u32 length=clink->header->length;
385 
386  for(i=0;i < 2;i++){
387  if((clink=apaDeleteFixPrev(clink, &rv))==NULL)
388  return 0;
389  if((clink=apaDeleteFixNext(clink, &rv))==NULL)
390  return 0;
391  }
392  if(clink->header->start==start && clink->header->length==length)
393  {
394  apaMakeEmpty(clink);
395  apaCacheFlushAllDirty(clink->device);
396  }
397  }
398  apaCacheFree(clink);
399  return rv;
400 }
