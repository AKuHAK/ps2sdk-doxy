102 void trap(exception_type_t type, struct exception_frame *ex) {
103  u32 i;
104  if(dbg_jmp_buf_setup) {
105  u32 *p = (u32*)dbg_jmp_buf;
106  /* simulate longjmp */
107  ex->epc = p[JUMP_BUF_PC];
108  ex->regs[29] = p[JUMP_BUF_SP];
109  ex->regs[30] = p[JUMP_BUF_FP];
110  ex->regs[16] = p[JUMP_BUF_S0];
111  ex->regs[17] = p[JUMP_BUF_S1];
112  ex->regs[18] = p[JUMP_BUF_S2];
113  ex->regs[19] = p[JUMP_BUF_S3];
114  ex->regs[20] = p[JUMP_BUF_S4];
115  ex->regs[21] = p[JUMP_BUF_S5];
116  ex->regs[22] = p[JUMP_BUF_S6];
117  ex->regs[23] = p[JUMP_BUF_S7];
118  ex->regs[28] = p[JUMP_BUF_GP];
119  ex->regs[2] = type; /* return value from setjmp */
120  return;
121  }
122  TRAP_PRINTF("IOP Exception : %s\n", exception_type_name[type]);
123  TRAP_PRINTF("EPC=%08x CAUSE=%08x SR=%08x BADVADDR=%08x DCIC=%08x\n", ex->epc, ex->cause, ex->sr, ex->badvaddr, ex->dcic);
124  for(i = 0; i != 32; i += 4) {
125  TRAP_PRINTF("r[%02d]=%08x r[%02d]=%08x r[%02d]=%08x r[%02d]=%08x",
126  i, ex->regs[i], i+1, ex->regs[i+1], i+2, ex->regs[i+2], i+3, ex->regs[i+3]);
127  }
128 
129  if (handlers[type]) {
130  handlers[type](type, ex);
131  }
132 
133  if(type == EXCEPTION_Bp) {
134  ex->dcic = 0;
135  } else {
136  if(ex->cause & (1<<31)) {
137  ex->cause &= ~(1<<31); /* clear BD */
138  } else {
139  ex->epc += 4;
140  }
141  }
142 }
