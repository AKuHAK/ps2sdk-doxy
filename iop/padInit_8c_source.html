233 static void DmaSendEE(void)
234 {
235  int dma_stat;
236 
237  dma_stat = sceSifDmaStat(sifdma_id);
238 
239  if(dma_stat >= 0)
240  {
241  if( (frame_count % 240) == 0)
242  {
243  M_KPRINTF("DMA Busy ID = %08x ret = %d\n", sifdma_id, dma_stat);
244  M_KPRINTF(" SB_STAT = %08x\n", SB_STAT);
245  }
246  }
247  else
248  {
249  u32 port, slot;
250  u32 sifdma_count = 1;
251 
252  /* This is where the 128*2 bytes of 'garbage' gets sent to EE.
253  I believe that only 16 bytes should have been sent, used for checking
254  which ports/slots are open on the EE (padGetConnection). However
255  someone made a mistake and sent 128 bytes instead :-) Its the first
256  112 (128-16) bytes of padState[0][0] which gets sent along with the
257  sif_buffer, which I think clearly indicates that sending 128 bytes
258  is an error. - Lukasz
259  */
260 
261  sif_buffer[0]++;
262  sif_buffer[1] = pad_portdata[0];
263  sif_buffer[2] = pad_portdata[1];
264 
265  if( (sif_buffer[0] % 2) == 0)
266  sifdma_td[0].dest = pad_ee_addr + 128;
267  else
268  sifdma_td[0].dest = pad_ee_addr;
269 
270  sifdma_td[0].src = sif_buffer;
271  sifdma_td[0].size = 128;
272  sifdma_td[0].attr = 0;
273 
274  for(port=0; port < 2; port++)
275  {
276  for(slot=0; slot < 4; slot++)
277  {
278  if( (openSlots[port] >> slot) & 1 )
279  {
280  padState_t *p = &padState[port][slot];
281 
282  // Setup EE pad data
283  p->ee_pdata.frame = p->frame;
284  p->ee_pdata.findPadRetries = p->findPadRetries;
285  p->ee_pdata.modeConfig = p->modeConfig;
286  p->ee_pdata.modeCurId = p->modeCurId;
287  p->ee_pdata.model = p->model;
288  p->ee_pdata.buttonDataReady = p->buttonDataReady;
289  p->ee_pdata.nrOfModes = p->numModes;
290  p->ee_pdata.modeCurOffs = p->modeCurOffs;
291  p->ee_pdata.nrOfActuators = p->numActuators;
292  p->ee_pdata.numActComb = p->numActComb;
293  p->ee_pdata.val_c6 = p->val_c6;
294  p->ee_pdata.mode = p->mode;
295  p->ee_pdata.lock = p->lock;
296  p->ee_pdata.actDirSize = p->ee_actDirectSize;
297  p->ee_pdata.state = p->state;
298  p->ee_pdata.reqState = p->reqState;
299  p->ee_pdata.currentTask = p->currentTask;
300 
301  p->frame++;
302 
303  p->ee_pdata.runTask = p->runTask;
304 
305  p->ee_pdata.actDirData[0] = p->ee_actDirectData.data32[0];
306  p->ee_pdata.actDirData[1] = p->ee_actDirectData.data32[1];
307  p->ee_pdata.actAlignData[0] = p->ee_actAlignData.data32[0];
308  p->ee_pdata.actAlignData[1] = p->ee_actAlignData.data32[1];
309 
310  p->ee_pdata.actData[0] = p->actData.data32[0];
311  p->ee_pdata.actData[1] = p->actData.data32[1];
312  p->ee_pdata.actData[2] = p->actData.data32[2];
313  p->ee_pdata.actData[3] = p->actData.data32[3];
314  p->ee_pdata.combData[0] = p->combData.data32[0];
315  p->ee_pdata.combData[1] = p->combData.data32[1];
316  p->ee_pdata.combData[2] = p->combData.data32[2];
317  p->ee_pdata.combData[3] = p->combData.data32[3];
318 
319  p->ee_pdata.modeTable[0] = p->modeTable.data32[0];
320  p->ee_pdata.modeTable[1] = p->modeTable.data32[1];
321 
322  p->ee_pdata.stat70bit = p->stat70bit;
323 
324  if( p->buttonDataReady == 1)
325  {
326  p->ee_pdata.length = padSetupEEButtonData(port, slot, p);
327  }
328  else
329  {
330  p->ee_pdata.length = 0;
331  }
332 
333  if( (p->frame & 1) == 0)
334  sifdma_td[sifdma_count].dest = (void*)p->padarea_ee_addr;
335  else
336  sifdma_td[sifdma_count].dest = (void*)(p->padarea_ee_addr + 128);
337 
338  sifdma_td[sifdma_count].src = &p->ee_pdata;
339  sifdma_td[sifdma_count].size = 128;
340  sifdma_td[sifdma_count].attr = 0;
341 
342  sifdma_count++;
343  }
344  }
345  }
346 
347  if(sifdma_count != 0)
348  {
349  int intr_state;
350 
351  CpuSuspendIntr(&intr_state);
352 
353  sifdma_id = sceSifSetDma( sifdma_td, sifdma_count);
354 
355  CpuResumeIntr(intr_state);
356 
357  if(sifdma_id == 0)
358  M_KPRINTF("sceSifSetDma failed\n");
359  }
360  }

414 static void MainThread(void *arg)
415 {
416  while(1)
417  {
418  u32 port, slot;
419 
420  mainThreadCount++;
421 
422  if( mainThreadCount % 30 == 0 ) sio2_mtap_update_slots();
423 
424  for(port=0; port < 2; port++)
425  {
426  for(slot=0; slot < 4; slot++)
427  {
428  if( ((openSlots[port] >> slot) & 0x1) == 1)
429  {
430  pdSetActive(port, slot, 0);
431 
432  padState[port][slot].stat70bit = pdGetStat70bit(port, slot);
433 
434  if(padState[port][slot].runTask != TASK_NONE)
435  {
436  if(padState[port][slot].runTask == TASK_PORT_CLOSE)
437  {
438  padState[port][slot].currentTask = padState[port][slot].runTask;
439  padState[port][slot].runTask = TASK_NONE;
440  padState[port][slot].reqState = PAD_RSTAT_BUSY;
441 
442  SetEventFlag(padState[port][slot].eventflag, EF_EXIT_THREAD);
443  }
444  else
445  {
446  if(padState[port][slot].currentTask == TASK_UPDATE_PAD)
447  {
448  // Start Task
449  StartThread(padState[port][slot].taskTid, NULL);
450  padState[port][slot].currentTask = padState[port][slot].runTask;
451  padState[port][slot].runTask = TASK_NONE;
452  padState[port][slot].reqState = PAD_RSTAT_BUSY;
453  }
454  else
455  {
456  padState[port][slot].runTask = TASK_NONE;
457  padState[port][slot].reqState = PAD_RSTAT_FAILED;
458  }
459  }
460  }
461  }
462 
463  switch(padState[port][slot].currentTask)
464  {
465  case TASK_UPDATE_PAD:
466  {
467  SetEventFlag(padState[port][slot].eventflag, EF_UPDATE_PAD);
468  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
469  pdSetActive(port, slot, 1);
470  } break;
471 
472  case TASK_QUERY_PAD:
473  {
474  padState[port][slot].buttonDataReady = 0;
475 
476  SetEventFlag(padState[port][slot].eventflag, EF_QUERY_PAD);
477  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
478  pdSetActive(port, slot, 1);
479  } break;
480 
481  case TASK_PORT_CLOSE:
482  {
483  if(GetThreadsStatus( &padState[port][slot] ) == 1)
484  {
485  padState[port][slot].currentTask = TASK_NONE;
486  padState[port][slot].reqState = PAD_RSTAT_COMPLETE;
487  openSlots[port] ^= (1 << slot);
488 
489  DeleteThreads( &padState[port][slot] );
490  SetEventFlag(padState[port][slot].eventflag, EF_PORT_CLOSE);
491  }
492 
493  } break;
494 
495  case TASK_SET_MAIN_MODE:
496  {
497  padState[port][slot].buttonDataReady = 0;
498 
499  SetEventFlag(padState[port][slot].eventflag, EF_SET_MAIN_MODE);
500  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
501  pdSetActive(port, slot, 1);
502  } break;
503 
504  case TASK_SET_ACT_ALIGN:
505  {
506  padState[port][slot].buttonDataReady = 0;
507 
508  SetEventFlag(padState[port][slot].eventflag, EF_SET_ACT_ALIGN);
509  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
510  pdSetActive(port, slot, 1);
511  } break;
512 
513  case TASK_SET_BUTTON_INFO:
514  {
515  padState[port][slot].buttonDataReady = 0;
516 
517  SetEventFlag(padState[port][slot].eventflag, EF_SET_SET_BUTTON_INFO);
518  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
519  pdSetActive(port, slot, 1);
520  } break;
521 
522  case TASK_SET_VREF_PARAM:
523  {
524  padState[port][slot].buttonDataReady = 0;
525 
526  SetEventFlag(padState[port][slot].eventflag, EF_SET_VREF_PARAM);
527  WaitEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_START, 0x10, 0);
528  pdSetActive(port, slot, 1);
529  } break;
530 
531  }
532  }
533  }
534 
535  // Transfer is started in VblankStart
536  vblankData.stopTransfer = 0;
537  WaitClearEvent(vblankData.eventflag, EF_VB_TRANSFER_DONE, WEF_AND|WEF_CLEAR, NULL);
538 
539  if( (openSlots[0] != 0) || (openSlots[1] != 0))
540  {
541  for(port=0; port < 2; port++)
542  {
543  for(slot=0; slot < 4; slot++)
544  {
545  if(pdIsActive(port, slot) == 1)
546  {
547  /* Signal transfer done and wait to task (reading
548  sio2 data) to be done. */
549  SetEventFlag(padState[port][slot].eventflag, EF_PAD_TRANSFER_DONE);
550  WaitEventFlag(padState[port][slot].eventflag, EF_TASK_DONE, 0x10, 0);
551  }
552  }
553  }
554  }
555 
556  // Send pad data to EE
557  DmaSendEE();
558 
559  mainThreadCount2++; // s7
560 
561  // Check for disconnected controllers
562  if(mainThreadCount2 >= 8)
563  {
564  if(mainThreadCount % 30 != 0)
565  {
566  if( pdIsActive(pad_port, pad_slot) == 1)
567  {
568  if( padState[pad_port][pad_slot].state == PAD_STATE_DISCONN)
569  pad_portdata[pad_port] &= ~(1 << pad_slot); // clear slot
570  else
571  pad_portdata[pad_port] |= (1 << pad_slot); // set slot
572  }
573  else
574  {
575  if( pdCheckConnection(pad_port, pad_slot) == 1)
576  pad_portdata[pad_port] |= (1 << pad_slot); // set slot
577  else
578  pad_portdata[pad_port] &= ~(1 << pad_slot); // clear slot
579  }
580 
581  //Move onto the next slot
582  pad_slot++;
583  if(pad_slot >= 4)
584  {
585  //Move onto the next port
586  pad_slot = 0;
587  pad_port++;
588 
589  if(pad_port >= 2) pad_port = 0;
590  }
591 
592  mainThreadCount2 = 0;
593  }
594  }
595  }
596 }
