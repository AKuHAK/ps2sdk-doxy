92 apa_cache_t *hddAddPartitionHere(s32 device, const apa_params_t *params, u32 *emptyBlocks,
93  u32 sector, int *err)
94 {
95  apa_cache_t *clink_this;
96  apa_cache_t *clink_next;
97  apa_cache_t *clink_new;
98  apa_header_t *header;
99  u32 i;
100  u32 tmp, some_size, part_end;
101  u32 tempSize;
102 
103  // walk empty blocks in case can use one :)
104  for(i=0;i< 32;i++)
105  {
106  if((u32)(1 << i) >= params->size && emptyBlocks[i]!=0)
107  return apaInsertPartition(device, params, emptyBlocks[i], err);
108  }
109  clink_this=apaCacheGetHeader(device, sector, APA_IO_MODE_READ, err);
110  header=clink_this->header;
111  part_end=header->start+header->length;
112  some_size=(part_end%params->size);
113  tmp = some_size ? params->size - some_size : 0;
114 
115  if(hddDevices[device].totalLBA < (part_end + params->size + tmp)
116  //Non-SONY: when dealing with large disks, this check may overflow (therefore, check for overflows!).
117  || (part_end < sector))
118  {
119  *err=-ENOSPC;
120  apaCacheFree(clink_this);
121  return NULL;
122  }
123 
124  if((clink_next=apaCacheGetHeader(device, 0, APA_IO_MODE_READ, err))==NULL)
125  {
126  apaCacheFree(clink_this);
127  return NULL;
128  }
129 
130  tempSize=params->size;
131  while(part_end%params->size)
132  {
133  tempSize=params->size>>1;
134  while(0x3FFFF<tempSize)
135  {
136  if(!(part_end%tempSize))
137  {
138  clink_new=apaRemovePartition(device, part_end, 0,
139  clink_this->header->start, tempSize);
140  clink_this->header->next=part_end;
141  clink_this->flags|=APA_CACHE_FLAG_DIRTY;
142  clink_next->header->prev=clink_new->header->start;
143  part_end+=tempSize;
144  clink_next->flags|=APA_CACHE_FLAG_DIRTY;
145  apaCacheFlushAllDirty(device);
146  apaCacheFree(clink_this);
147  clink_this=clink_new;
148  break;
149  }
150  tempSize>>=1;
151  }
152  }
153  if((clink_new=apaFillHeader(device, params, part_end, 0, clink_this->header->start,
154  params->size, err))!=NULL)
155  {
156  clink_this->header->next=part_end;
157  clink_this->flags|=APA_CACHE_FLAG_DIRTY;
158  clink_next->header->prev=clink_new->header->start;
159  clink_next->flags|=APA_CACHE_FLAG_DIRTY;
160  apaCacheFlushAllDirty(device);
161  }
162  apaCacheFree(clink_this);
163  apaCacheFree(clink_next);
164  return clink_new;
165 }

298 #else

300 #endif
301  return MODULE_RESIDENT_END;
302  }
