100 static int fioGetInput(const char *arg, apa_params_t *params)
101 {
102  char argBuf[32];
103  int rv=0, i;
104  static const struct apaFsType fsTypes[]={
105  {"PFS", APA_TYPE_PFS},
106  {"CFS", APA_TYPE_CFS},
107  {"EXT2", APA_TYPE_EXT2},
108  {"EXT2SWAP", APA_TYPE_EXT2SWAP}
109  };
110 
111  if(params==NULL)
112  return -EINVAL;
113  memset(params, 0, sizeof(apa_params_t));
114 
115  while(arg[0]==' ') arg++;
116 
117  if(arg[0]==0 || arg[0]==',')
118  return -EINVAL;
119  if((rv=fioInputBreaker(&arg, params->id, APA_IDMAX))!=0)
120  return rv;
121  if(arg[0] == '\0') // Return if there are no further parameters.
122  return 0;
123 
124  if((rv=fioInputBreaker(&arg, params->fpwd, APA_PASSMAX))!=0)
125  return rv;
126 
127  if(params->fpwd[0] != '\0')
128  apaEncryptPassword(params->id, params->fpwd, params->fpwd);
129 
130  if(arg[0] == '\0') // Return if there are no further parameters.
131  return 0;
132 
133  if((rv=fioInputBreaker(&arg, params->rpwd, APA_PASSMAX))!=0)
134  return rv;
135 
136  if(params->rpwd[0] != '\0')
137  apaEncryptPassword(params->id, params->rpwd, params->rpwd);
138 
139  if(arg[0] == '\0') // Return if there are no further parameters.
140  return 0;
141 
142  memset(argBuf, 0, sizeof(argBuf));
143  if((rv=fioInputBreaker(&arg, argBuf, sizeof(argBuf)))!=0)
144  return rv;
145 
146  if((rv=fioPartitionSizeLookUp(argBuf))<0)
147  return rv;
148  params->size=rv;
149 
150  memset(argBuf, 0, sizeof(argBuf));
151  if((rv=fioInputBreaker(&arg, argBuf, sizeof(argBuf)))!=0)
152  return rv;
153 
154  for(i = 0; i < 4; i++)
155  {
156  if(!strcmp(argBuf, fsTypes[i].desc))
157  {
158  params->type = fsTypes[i].type;
159  break;
160  }
161  }
162 
163  if(i == 4)
164  {
165  printf("hdd: error: Invalid fstype, %s.\n", argBuf);
166  return -EINVAL;
167  }
168 
169  return rv;
170 }

192 static int fioDataTransfer(iop_file_t *f, void *buf, int size, int mode)
193 {
194  hdd_file_slot_t *fileSlot=(hdd_file_slot_t *)f->privdata;
195 
196  if((size & 0x1FF))
197  return -EINVAL;
198  size>>=9; // size/512
199 
200  if(fileSlot->post+size>=0x1FF9)// no over reading
201  size=0x1FF8-fileSlot->post;
202 
203  if(size!=0) {
204  int rv=0;
205 
206  WaitSema(fioSema);
207  if(ata_device_sector_io(f->unit, buf, fileSlot->post+fileSlot->parts[0].start+8, size, mode))
208  rv=-EIO;
209  SignalSema(fioSema);
210  if(rv==0)
211  {
212  fileSlot->post+=size;
213  return size<<9;
214  }
215 
216  return rv;
217  }
218  return 0;
219 }

345 static int apaOpen(s32 device, hdd_file_slot_t *fileSlot, apa_params_t *params, int mode)
346 {
347  int rv=0;
348  u32 emptyBlocks[32];
349  apa_cache_t *clink;
350  apa_cache_t *clink2;
351  u32 sector=0;
352 
353  // walk all looking for any empty blocks & look for partition
354  clink=apaCacheGetHeader(device, 0, APA_IO_MODE_READ, &rv);
355  memset(&emptyBlocks, 0, sizeof(emptyBlocks));
356  while(clink)
357  {
358  sector=clink->sector;
359  if(!(clink->header->flags & APA_FLAG_SUB)) {
360  if(memcmp(clink->header->id, params->id, APA_IDMAX) == 0)
361  break; // found :)
362  }
363  apaAddEmptyBlock(clink->header, emptyBlocks);
364  clink=apaGetNextHeader(clink, &rv);
365  }
366 
367  if(rv!=0)
368  return rv;
369  rv=-ENOENT;
370 
371  if(clink==NULL && (mode & O_CREAT))
372  {
373  if((rv=hddCheckPartitionMax(device, params->size))>=0) {
374  if((clink=hddAddPartitionHere(device, params, emptyBlocks, sector, &rv))!=NULL)
375  {
376  sector=clink->header->start;
377  clink2=apaCacheAlloc();
378  memset(clink2->header, 0, sizeof(apa_header_t));
379  ata_device_sector_io(device, clink2->header, sector+8 , 2, ATA_DIR_WRITE);
380  ata_device_sector_io(device, clink2->header, sector+0x2000, 2, ATA_DIR_WRITE);
381  apaCacheFree(clink2);
382  }
383  }
384  }
385  if(clink==NULL)
386  return rv;
387  fileSlot->parts[0].start=clink->header->start;
388  fileSlot->parts[0].length=clink->header->length;
389  memcpy(&fileSlot->parts[1], &clink->header->subs, APA_MAXSUB*sizeof(apa_sub_t));
390  fileSlot->type=clink->header->type;
391  fileSlot->nsub=clink->header->nsub;
392  memcpy(&fileSlot->id, &clink->header->id, APA_IDMAX);
393  apaCacheFree(clink);
394  if(apaPassCmp(clink->header->fpwd, params->fpwd)!=0)
395  {
396  rv = (!(mode & O_WRONLY)) ? apaPassCmp(clink->header->rpwd, params->rpwd) : -EACCES;
397  } else
398  rv = 0;
399 
400  return rv;
401 }

403 static int apaRemove(s32 device, const char *id, const char *fpwd)
404 {
405  u32 nsub;
406  apa_cache_t *clink;
407  apa_cache_t *clink2;
408  int rv, i;
409 
410  for(i=0;i<apaMaxOpen;i++) // look to see if open
411  {
412  if(hddFileSlots[i].f!=0) {
413  if(memcmp(hddFileSlots[i].id, id, APA_IDMAX)==0)
414  return -EBUSY;
415  }
416  }
417  if(id[0]=='_' && id[1]=='_')
418  return -EACCES;
419  if((clink=apaFindPartition(device, id, &rv))==NULL)
420  return rv;
421  if(apaPassCmp(clink->header->fpwd, fpwd))
422  {
423  apaCacheFree(clink);

425  }
426  // remove all subs frist...
427  nsub=clink->header->nsub;
428  clink->header->nsub=0;
429  clink->flags|=APA_CACHE_FLAG_DIRTY;
430  apaCacheFlushAllDirty(device);
431  for(i=nsub-1;i!=-1;i--)
432  {
433  if((clink2=apaCacheGetHeader(device, clink->header->subs[i].start, APA_IO_MODE_READ, &rv))){
434  if((rv=apaDelete(clink2))){
435  apaCacheFree(clink);
436  return rv;
437  }
438  }
439  }
440  if(rv==0)
441  return apaDelete(clink);
442 
443  apaCacheFree(clink);
444  return rv;
445 }

467  {
468  SignalSema(fioSema);
469  return -EBUSY;
470  }
471  }
472  }
473

475  if(oldParams->id[0]=='_' && oldParams->id[1]=='_')
476  return -EACCES;
477 
478  // find :)
479  if((clink = apaFindPartition(device, oldParams->id, &rv)) == NULL)
480  {
481  SignalSema(fioSema);
482  return rv;
483  }
484 
485  // Check for access rights.
486  if(apaPassCmp(clink->header->fpwd, oldParams->fpwd) != 0)
487  {
488  apaCacheFree(clink);
489  return -EACCES;
490  }
491 
492  // do the renaming :) note: subs have no names!!
493  memcpy(clink->header->id, newParams->id, APA_IDMAX);
494 
495  // Update passwords
496  memcpy(clink->header->rpwd, newParams->rpwd, APA_PASSMAX);
497  memcpy(clink->header->fpwd, newParams->fpwd, APA_PASSMAX);
498 
499  clink->flags|=APA_CACHE_FLAG_DIRTY;
500 
501  apaCacheFlushAllDirty(device);
502  apaCacheFree(clink);
503 
504  return 0;
505 }

733 static int ioctl2AddSub(hdd_file_slot_t *fileSlot, char *argp)
734 {
735  int rv;
736  u32 device=fileSlot->f->unit;
737  apa_params_t params;
738  u32 emptyBlocks[32];
739  apa_cache_t *clink;
740  u32 sector=0;
741  u32 length;
742 
743  if(!(fileSlot->f->mode & O_WRONLY))
744  return -EACCES;
745 
746  if(!(fileSlot->nsub < APA_MAXSUB))
747  return -EFBIG;
748 
749  memset(&params, 0, sizeof(apa_params_t));
750 
751  if((rv=fioPartitionSizeLookUp(argp)) < 0)
752  return rv;
753 
754  params.size=rv;
755  params.flags=APA_FLAG_SUB;
756  params.type=fileSlot->type;
757  params.main=fileSlot->parts[0].start;
758  params.number=fileSlot->nsub+1;
759  if((rv=hddCheckPartitionMax(device, params.size)) < 0)
760  return rv;
761 
762  // walk all looking for any empty blocks
763  memset(&emptyBlocks, 0, sizeof(emptyBlocks));
764  clink=apaCacheGetHeader(device, 0, APA_IO_MODE_READ, &rv);
765  while(clink){
766  sector=clink->sector;
767  apaAddEmptyBlock(clink->header, emptyBlocks);
768  clink=apaGetNextHeader(clink, &rv);
769  }
770  if(rv!=0)
771  return rv;
772 
773  if(!(clink=hddAddPartitionHere(device, &params, emptyBlocks, sector, &rv)))
774  return rv;
775 
776  sector=clink->header->start;
777  length=clink->header->length;
778  apaCacheFree(clink);
779  if(!(clink=apaCacheGetHeader(device, fileSlot->parts[0].start, APA_IO_MODE_READ, &rv)))
780  return rv;
781 
782  clink->header->subs[clink->header->nsub].start=sector;
783  clink->header->subs[clink->header->nsub].length=length;
784  clink->header->nsub++;
785  fileSlot->nsub++;
786  fileSlot->parts[fileSlot->nsub].start=sector;
787  fileSlot->parts[fileSlot->nsub].length=length;
788  clink->flags|=APA_CACHE_FLAG_DIRTY;
789  apaCacheFlushAllDirty(device);
790  apaCacheFree(clink);
791  return rv;
