95 ComputeTimeDiff(iop_sys_clock_t* pStart,iop_sys_clock_t* pEnd)
96 {
97  iop_sys_clock_t Diff;
98  u32 iSec, iUSec, iDiff;
99 
100  Diff.lo=pEnd->lo-pStart->lo;
101  Diff.hi=pEnd->hi-pStart->hi - (pStart->lo>pEnd->lo);
102 
103  SysClock2USec(&Diff, &iSec, &iUSec);
104  iDiff=(iSec*1000)+(iUSec/1000);
105 
106  return((iDiff!=0)?iDiff:1);

194 static u32_t sys_arch_mbox_fetch_internal(sys_mbox_t pMBox, void** ppvMSG, u32_t TimeElaspedout, char block)
195 {
196  void *pmsg;
197  u32_t TimeElasped = 0;
198  iop_sys_clock_t ClockTicks;
199  iop_sys_clock_t Start;
200  iop_sys_clock_t End;
201  int result, iPID;
202 
203  if(block){
204  iPID=GetThreadId();
205 
206  if(TimeElaspedout > 0)
207  {
208  GetSystemTime(&Start);
209  USec2SysClock(TimeElaspedout * 1000, &ClockTicks);
210  SetAlarm(&ClockTicks, &TimeoutHandler, (void*)iPID);
211  }
212 
213  if((result=ReceiveMbx(&pmsg, pMBox))!= 0) { return(SYS_ARCH_TIMEOUT); }
214 
215  if(TimeElaspedout > 0)
216  {
217  CancelAlarm(TimeoutHandler,(void*)iPID);
218  GetSystemTime(&End);
219 
220  TimeElasped = ComputeTimeDiff(&Start, &End);
221  }
222  }
223  else{
224  TimeElasped=((result=PollMbx(&pmsg, pMBox))!=0)?SYS_MBOX_EMPTY:0;
225  }
226 
227  if(result==0){
228  *ppvMSG = ((arch_message *)pmsg)->sys_msg;
229  free_msg((arch_message *) pmsg);
230  }
231 
232  //Return the number of msec waited.
233  return TimeElasped;
234 }

271 u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t timeout)
272 {
273  u32_t result;
274 
275  //Wait TimeElaspedout msec for the Sema to receive a signal.
276  if(timeout==0)
277  {
278  //Wait with no timeouts.
279  result=(WaitSema(*sem)==0)?0:SYS_ARCH_TIMEOUT;
280  }
281  else if(timeout==1)
282  {
283  //Poll.
284  result=(PollSema(*sem)==0)?0:SYS_ARCH_TIMEOUT;
285  }
286  else
287  {
288  //Use alarm to timeout.
289  iop_sys_clock_t ClockTicks;
290  iop_sys_clock_t Start;
291  iop_sys_clock_t End;
292  int iPID=GetThreadId();
293  u32_t u32WaitTime;
294 
295  GetSystemTime(&Start);
296  USec2SysClock(timeout*1000,&ClockTicks);
297  SetAlarm(&ClockTicks, &TimeoutHandler, (void*)iPID);
298 
299  if(WaitSema(*sem)==0){
300  CancelAlarm(TimeoutHandler,(void*)iPID);
301  GetSystemTime(&End);
302 
303  u32WaitTime=ComputeTimeDiff(&Start, &End);
304  result=(u32WaitTime<=timeout)?u32WaitTime:timeout;
305  }else result=SYS_ARCH_TIMEOUT;
306  }
307 
308  return result;
309 }
