95 static int fat_getClusterChain12(fat_driver* fatd, unsigned int cluster, unsigned int* buf, unsigned int bufSize, int startFlag) {
96  int ret;
97  unsigned int i, recordOffset, fatSector, lastFatSector;
98  unsigned char xbuf[4], sectorSpan, cont;
99  unsigned char* sbuf = NULL; //sector buffer
100 
101  cont = 1;
102  lastFatSector = -1;
103  i = 0;
104  if (startFlag) {
105  buf[i] = cluster; //store first cluster
106  i++;
107  }
108  while(i < bufSize && cont) {
109  recordOffset = (cluster * 3) / 2; //offset of the cluster record (in bytes) from the FAT start
110  fatSector = recordOffset / fatd->partBpb.sectorSize;
111  sectorSpan = 0;
112  if ((recordOffset % fatd->partBpb.sectorSize) == (fatd->partBpb.sectorSize - 1)) {
113  sectorSpan = 1;
114  }
115  if (lastFatSector != fatSector || sectorSpan) {
116  ret = READ_SECTOR(fatd->dev, fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector, sbuf);
117  if (ret < 0) {
118  XPRINTF("USBHDFSD: Read fat12 sector failed! sector=%u! \n", fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector );
119  return -EIO;
120  }
121  lastFatSector = fatSector;
122 
123  if (sectorSpan) {
124  xbuf[0] = sbuf[fatd->partBpb.sectorSize - 2];
125  xbuf[1] = sbuf[fatd->partBpb.sectorSize - 1];
126  ret = READ_SECTOR(fatd->dev, fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector + 1, sbuf);
127  if (ret < 0) {
128  XPRINTF("USBHDFSD: Read fat12 sector failed sector=%u! \n", fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector + 1);
129  return -EIO;
130  }
131  xbuf[2] = sbuf[0];
132  xbuf[3] = sbuf[1];
133  }
134  }
135  if (sectorSpan) { // use xbuf as source buffer
136  cluster = fat_getClusterRecord12(xbuf + (recordOffset % fatd->partBpb.sectorSize) - (fatd->partBpb.sectorSize-2), cluster % 2);
137  } else { // use sector buffer as source buffer
138  cluster = fat_getClusterRecord12(sbuf + (recordOffset % fatd->partBpb.sectorSize), cluster % 2);
139  }
140 
141  if ((cluster & 0xFFF) >= 0xFF8) {
142  cont = 0; //continue = false
143  } else {
144  buf[i] = cluster & 0xFFF;
145  i++;
146  }
147  }
148  return i;
149 }

351  }
352  //detect deleted entry - it will be ignored
353  if (dir_entry->sfn.name[0] == 0xE5) {
354  return 3;
355  }
356 
357  //detect long filename
358  if (dir_entry->lfn.rshv == 0x0F && dir_entry->lfn.reserved1 == 0x00 && dir_entry->lfn.reserved2[0] == 0x00) {
359  //long filename - almost whole direntry is unicode string - extract it
360  offset = dir_entry->lfn.entrySeq & 0x3f;
361  offset--;
362  offset = offset * 13;
363  //name - 1st part
364  cont = 1;
365  for (i = 0; i < 10 && cont; i+=2) {
366  character = dir_entry->lfn.name1[i] | (dir_entry->lfn.name1[i+1] << 8);
367 
368  if (character == 0 || offset >= FAT_MAX_NAME) {
369  dir->name[offset] = 0; //terminate
370  cont = 0; //stop
371  } else {
372  // Handle non-ASCII characters
373  dir->name[offset] = character < 128 ? dir_entry->lfn.name1[i] : '?';
374  offset++;
375  }
376  }
377  //name - 2nd part
378  for (i = 0; i < 12 && cont; i+=2) {
379  character = dir_entry->lfn.name2[i] | (dir_entry->lfn.name2[i+1] << 8);
380 
381  if (character == 0 || offset >= FAT_MAX_NAME) {
382  dir->name[offset] = 0; //terminate
383  cont = 0; //stop
384  } else {
385  // Handle non-ASCII characters
386  dir->name[offset] = character < 128 ? dir_entry->lfn.name2[i] : '?';
387  offset++;
388  }
389  }
390  //name - 3rd part
391  for (i = 0; i < 4 && cont; i+=2) {
392  character = dir_entry->lfn.name3[i] | (dir_entry->lfn.name3[i+1] << 8);
393 
394  if (character == 0 || offset >= FAT_MAX_NAME) {
395  dir->name[offset] = 0; //terminate
396  cont = 0; //stop
397  } else {
398  // Handle non-ASCII characters
399  dir->name[offset] = character < 128 ? dir_entry->lfn.name3[i] : '?';
400  offset++;
401  }
402  }
403  if ((dir_entry->lfn.entrySeq & 0x40)) { //terminate string flag
404  dir->name[offset] = 0;
405  }
406  return 2;
407  } else {
408  //short filename
409  //copy name
410  for (i = 0; i < 8 && dir_entry->sfn.name[i]!= ' '; i++) {
411  dir->sname[i] = dir_entry->sfn.name[i];
412  // NTadaption for LaunchELF
413  if (dir_entry->sfn.reservedNT & 0x08 &&
414  dir->sname[i] >= 'A' && dir->sname[i] <= 'Z') {
415  dir->sname[i] += 0x20; //Force standard letters in name to lower case
416  }
417  }
418  for (j=0; j < 3 && dir_entry->sfn.ext[j] != ' '; j++) {
419  if (j == 0) {
420  dir->sname[i] = '.';
421  i++;
422  }
423  dir->sname[i+j] = dir_entry->sfn.ext[j];
424  // NTadaption for LaunchELF
425  if (dir_entry->sfn.reservedNT & 0x10 &&
426  dir->sname[i+j] >= 'A' && dir->sname[i+j] <= 'Z') {
427  dir->sname[i+j] += 0x20; //Force standard letters in ext to lower case
428  }
429  }

431  if (dir->name[0] == 0) { //long name desn't exit
432  for (i =0 ; dir->sname[i] !=0; i++) dir->name[i] = dir->sname[i];
433  dir->name[i] = 0;
434  }

737 int fat_readFile(fat_driver* fatd, fat_dir* fatDir, unsigned int filePos, unsigned char* buffer, unsigned int size) {
738  int ret, chainSize;
739  unsigned int i, j, startSector, clusterChainStart, bufSize, sectorSkip, clusterSkip, dataSkip;
740  unsigned char nextChain;
741  struct SBP2Device* mass_device = fatd->dev;
742 
743  unsigned int bufferPos, fileCluster, clusterPos;
744 
745  fat_getClusterAtFilePos(fatd, fatDir, filePos, &fileCluster, &clusterPos);
746  sectorSkip = (filePos - clusterPos) / fatd->partBpb.sectorSize;
747  clusterSkip = sectorSkip / fatd->partBpb.clusterSize;
748  sectorSkip %= fatd->partBpb.clusterSize;
749  dataSkip = filePos % fatd->partBpb.sectorSize;
750  bufferPos = 0;
751 
752  XPRINTF("USBHDFSD: fileCluster = %u, clusterPos= %u clusterSkip=%u, sectorSkip=%u dataSkip=%u \n",
753  fileCluster, clusterPos, clusterSkip, sectorSkip, dataSkip);
754 
755  if (fileCluster < 2) {
756  return 0;
757  }
758 
759  bufSize = mass_device->sectorSize;
760  nextChain = 1;
761  clusterChainStart = 1;
762 
763  while (nextChain && size > 0 ) {
764  if((chainSize = fat_getClusterChain(fatd, fileCluster, fatd->cbuf, MAX_DIR_CLUSTER, clusterChainStart))<0){
765  return chainSize;
766  }
767 
768  clusterChainStart = 0;
769  if (chainSize >= MAX_DIR_CLUSTER) { //the chain is full, but more chain parts exist
770  fileCluster = fatd->cbuf[MAX_DIR_CLUSTER - 1];
771  }else { //chain fits in the chain buffer completely - no next chain needed
772  nextChain = 0;
773  }
774  while (clusterSkip >= MAX_DIR_CLUSTER) {
775  chainSize = fat_getClusterChain(fatd, fileCluster, fatd->cbuf, MAX_DIR_CLUSTER, clusterChainStart);
776  clusterChainStart = 0;
777  if (chainSize >= MAX_DIR_CLUSTER) { //the chain is full, but more chain parts exist
778  fileCluster = fatd->cbuf[MAX_DIR_CLUSTER - 1];
779  }else { //chain fits in the chain buffer completely - no next chain needed
780  nextChain = 0;
781  }
782  clusterSkip -= MAX_DIR_CLUSTER;
783  }
784 
785  //process the cluster chain (fatd->cbuf) and skip leading clusters if needed
786  for (i = 0 + clusterSkip; i < chainSize && size > 0; i++) {
787  //read cluster and save cluster content
788  startSector = fat_cluster2sector(&fatd->partBpb, fatd->cbuf[i]);
789  //process all sectors of the cluster (and skip leading sectors if needed)
790  for (j = 0 + sectorSkip; j < fatd->partBpb.clusterSize && size > 0; j++) {
791  unsigned char* sbuf = NULL; //sector buffer
792 
793  ret = READ_SECTOR(fatd->dev, startSector + j, sbuf);
794  if (ret < 0) {
795  XPRINTF("USBHDFSD: Read sector failed ! sector=%u\n", startSector + j);
796  return bufferPos;
797  }
798 
799  //compute exact size of transfered bytes
800  if (size < bufSize) {
801  bufSize = size + dataSkip;
802  }
803  if (bufSize > mass_device->sectorSize) {
804  bufSize = mass_device->sectorSize;
805  }
806  XPRINTF("USBHDFSD: memcopy dst=%u, src=%u, size=%u bufSize=%u \n", bufferPos, dataSkip, bufSize-dataSkip, bufSize);
807  memcpy(buffer+bufferPos, sbuf + dataSkip, bufSize - dataSkip);
808  size-= (bufSize - dataSkip);
809  bufferPos += (bufSize - dataSkip);
810  dataSkip = 0;
811  bufSize = mass_device->sectorSize;
812  }
813  sectorSkip = 0;
814  }
815  clusterSkip = 0;
816  }
817  return bufferPos;
818 }
