100 void hubStatusChangeCallback(IoRequest *req) {
101  UsbHub *dev = (UsbHub *)req->userCallbackArg;
102  int port;
103 
104  if (req->resultCode == USB_RC_OK) {
105  if (dev->statusChangeInfo[0] & 1) {
106  dev->statusChangeInfo[0] &= ~1;
107  HubControlTransfer(dev,
108  USB_DIR_IN | USB_RT_HUB, USB_REQ_GET_STATUS, 0, 0, 4, &dev->hubStatus,
109  hubGetHubStatusCallback);
110  } else {
111  if (dev->hubStatusCounter > 0) {
112  port = dev->hubStatusCounter;
113  if (dev->statusChangeInfo[port >> 3] & BIT(port & 7)) {
114  dev->statusChangeInfo[port >> 3] &= ~BIT(port & 7);
115  dev->portCounter = port;
116  HubControlTransfer(dev,
117  USB_DIR_IN | USB_RT_PORT, USB_REQ_GET_STATUS, 0, port, 4, &dev->portStatusChange,
118  hubGetPortStatusCallback);
119  return;
120  }
121  } else {
122  for (port = 1; port <= dev->numChildDevices; port++) {
123  if (dev->statusChangeInfo[port >> 3] & BIT(port & 7)) {
124  dev->statusChangeInfo[port >> 3] &= ~BIT(port & 7);

129  return;

131  }

178 int addTimerCallback(TimerCbStruct *arg, TimerCallback func, void *cbArg, u32 delay) {
179  if (arg->isActive)
180  return -1;
181 
182  arg->isActive = 1;
183  arg->callbackProc = func;
184  arg->callbackArg = cbArg;
185 
186  TimerCbStruct *pos = memPool.timerListStart;
187 
188  while (pos && (delay > pos->delayCount)) {
189  delay -= pos->delayCount;
190  pos = pos->next;
191  }
192  if (pos) {
193  arg->prev = pos->prev;
194  if (pos->prev)
195  pos->prev->next = arg;
196  else
197  memPool.timerListStart = arg;
198 
199  arg->next = pos;
200  pos->prev = arg;
201  pos->delayCount -= delay;
202  } else {
203  arg->prev = memPool.timerListEnd;
204  if (memPool.timerListEnd)
205  memPool.timerListEnd->next = arg;
206  else
207  memPool.timerListStart = arg;
208  memPool.timerListEnd = arg;
209  arg->next = NULL;
210  }
211  arg->delayCount = delay;
212  memPool.ohciRegs->HcInterruptEnable = OHCI_INT_SF;
213  return 0;

382 void fetchConfigDescriptors(IoRequest *req) {
383  Endpoint *ep = req->correspEndpoint;
384  Device *dev = ep->correspDevice;
385  u16 readLen;
386  int fetchDesc;
387 
388  if ((req->resultCode == USB_RC_OK) || (dev->fetchDescriptorCounter == 0)) {
389  u32 curDescNum = dev->fetchDescriptorCounter++;
390 
391  fetchDesc = curDescNum & 1;
392  curDescNum >>= 1;
393 
394  if ((curDescNum > 0) && !fetchDesc) {
395  UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
396  dev->staticDeviceDescEndPtr += READ_UINT16(&desc->wTotalLength);
397  }
398 
399  if (fetchDesc) {
400  UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
401  readLen = READ_UINT16(&desc->wTotalLength);
402  } else
403  readLen = 4;
404 
405  if ((u8*)dev->staticDeviceDescEndPtr + readLen > (u8*)dev->staticDeviceDescPtr + usbConfig.maxStaticDescSize) {
406  dbg_printf("USBD: Device ignored, Device descriptors too large\n");
407  return; // buffer is too small, silently ignore the device
408  }
409 
410  if (curDescNum < ((UsbDeviceDescriptor*)dev->staticDeviceDescPtr)->bNumConfigurations) {
411  doControlTransfer(ep, &dev->ioRequest,
412  USB_DIR_IN | USB_RECIP_DEVICE, USB_REQ_GET_DESCRIPTOR, (USB_DT_CONFIG << 8) | curDescNum, 0, readLen,
413  dev->staticDeviceDescEndPtr, fetchConfigDescriptors);
414  } else
415  connectNewDevice(dev);
416  } else
417  killDevice(dev, ep);
418 }

495 void hubGetPortStatusCallback(IoRequest *req) {
496  int feature = -1;
497  UsbHub *dev = (UsbHub *)req->userCallbackArg;
498  Device *port;
499  if (req->resultCode == USB_RC_OK) {
500  dbg_printf("port status change: %d: %08X\n", dev->portCounter, dev->portStatusChange);
501  if (dev->portStatusChange & BIT(C_PORT_CONNECTION))
502  feature = C_PORT_CONNECTION;
503  else if (dev->portStatusChange & BIT(C_PORT_ENABLE))
504  feature = C_PORT_ENABLE;
505  else if (dev->portStatusChange & BIT(C_PORT_SUSPEND))
506  feature = C_PORT_SUSPEND;
507  else if (dev->portStatusChange & BIT(C_PORT_OVER_CURRENT))
508  feature = C_PORT_OVER_CURRENT;
509  else if (dev->portStatusChange & BIT(C_PORT_RESET))
510  feature = C_PORT_RESET;
511 
512  if (feature >= 0) {
513  dev->portStatusChange &= ~BIT(feature);
514  HubControlTransfer(dev,
515  USB_DIR_OUT | USB_RT_PORT, USB_REQ_CLEAR_FEATURE, feature, dev->portCounter, 0, NULL,
516  hubGetPortStatusCallback);
517  } else {
518  port = fetchPortElemByNumber(dev->controlEp->correspDevice, dev->portCounter);
519  if (port) {
520  if (dev->portStatusChange & BIT(PORT_CONNECTION)) {
521  dbg_printf("Hub Port CCS\n");
522  if (port->deviceStatus == DEVICE_NOTCONNECTED) {
523  dbg_printf("resetting dev\n");
524  port->deviceStatus = DEVICE_CONNECTED;
525  addTimerCallback(&port->timer, (TimerCallback)hubResetDevice, port, 500);
526  return;
527  } else if (port->deviceStatus == DEVICE_RESETPENDING) {
528  if (dev->portStatusChange & BIT(PORT_ENABLE)) {
529  dbg_printf("hub port reset done, opening control EP\n");
530  port->deviceStatus = DEVICE_RESETCOMPLETE;
531  port->isLowSpeedDevice = (dev->portStatusChange >> PORT_LOW_SPEED) & 1;
532 
533  if (openDeviceEndpoint(port, NULL, 0))
534  hubTimedSetFuncAddress(port);
535  else
536  dbg_printf("Can't open default control ep.\n");
537  dev->hubStatusCounter = 0;
538  }
539  }
540  } else {
541  dbg_printf("disconnected; flushing port\n");
542  flushPort(port);
543  }
544  }
545  hubStatusChangeCallback(&dev->statusIoReq);
546  }
547  } else
548  dbg_printf("HubGetPortStatusCallback res %d\n", req->resultCode);
549 }
