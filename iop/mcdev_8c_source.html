76 {
77  register int r = errcode;
78 
79  if (r < 0) {
80  switch (r + 8) {
81  case 0:
82  r = -EIO;
83  break;
84  case 1:
85  r = -EMFILE;

111 int mcman_modloadcb(char *filename, int *port, int *slot)
112 {
113  register char *path = filename;
114  register int upos;
115 
116  if (*path == 0x20) {
117  path++;
118  while (*path == 0x20)
119  path++;
120  }
121 
122  if (((u8)path[0] | 0x20) != mcman_mcdev.name[0])
123  return 0;
124  if (((u8)path[1] | 0x20) != mcman_mcdev.name[1])
125  return 0;
126 
127  if ((u32)strlen(path) < 2) return 2;
128 
129  upos = mcman_chrpos(path, ':');
130 
131  if (upos < 0)
132  upos = strlen(path);
133 
134  if (port) {
135  upos --;
136  if (((u8)path[upos] - 0x30) < 10)
137  *port = (u8)path[upos] - 0x30;
138  else *port = 0;
139  }
140 
141  upos--;
142  *port += 2;
143 
144  if (slot) {
145  register int m, v;
146 
147  *slot = 0;
148  for (m = 1; ((u8)path[upos --] - 0x30) < 10; m = v << 1) {
149  v = m << 2;
150  *slot += m * (path[upos] - 0x30);
151  v += m;
152  }
153  }
154 
155  return 1;
156 }

417 int mc_chstat(iop_file_t *f, char *filename, io_stat_t *stat, u32 statmask)
418 {
419  register int r, flags;
420  sceMcTblGetDir mctbl;
421 
422  WaitSema(mcman_io_sema);
423  mcman_unit2card(f->unit);
424 
425  r = McDetectCard2(mcman_mc_port, mcman_mc_slot);
426  if (r >= -1) {
427  if (statmask & SCE_CST_ATTR) {
428  flags = 0x008;
429  mctbl.Reserve2 = stat->attr;
430  }
431  else flags = 0x000;
432 
433  if (statmask & SCE_CST_MODE) {
434  flags |= 0x200;
435  if (stat->mode & SCE_STM_R) mctbl.AttrFile |= sceMcFileAttrReadable;
436  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrReadable;
437 
438  if (stat->mode & SCE_STM_W) mctbl.AttrFile |= sceMcFileAttrWriteable;
439  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrWriteable;
440 
441  if (stat->mode & SCE_STM_X) mctbl.AttrFile |= sceMcFileAttrExecutable;
442  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrExecutable;
443 
444  if (stat->mode & SCE_STM_C) mctbl.AttrFile |= sceMcFileAttrDupProhibit;
445  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrDupProhibit;
446 
447  if (stat->mode & sceMcFileAttrPS1) mctbl.AttrFile |= sceMcFileAttrPS1;
448  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrPS1;
449 
450  if (stat->mode & sceMcFileAttrPDAExec) mctbl.AttrFile |= sceMcFileAttrPDAExec;
451  else mctbl.AttrFile &= (unsigned short)~sceMcFileAttrPDAExec;
452  }
453 
454  if (statmask & SCE_CST_CT) {
455  flags |= 0x001;
456  memcpy(&mctbl._Create, stat->ctime, sizeof(sceMcStDateTime));
457  }
458 
459  if (statmask & SCE_CST_MT) {
460  flags |= 0x002;
461  memcpy(&mctbl._Modify, stat->mtime, sizeof(sceMcStDateTime));
462  }
463 
464  r = McSetFileInfo(mcman_mc_port, mcman_mc_slot, filename, &mctbl, flags);
465  }
466  SignalSema(mcman_io_sema);
467 
468  return mcman_ioerrcode(r);
469 }
