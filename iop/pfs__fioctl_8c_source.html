116 int pfsFioIoctl2(iop_file_t *f, int cmd, void *arg, size_t arglen, void *buf, size_t buflen)
117 {
118  int rv;
119  pfs_file_slot_t *fileSlot = (pfs_file_slot_t *)f->privdata;
120  pfs_mount_t *pfsMount;
121 
122  if(f->mode & O_DIROPEN)
123  if(cmd==PIOCATTRREAD)
124  return -EISDIR;
125 
126  if(!(f->mode & O_WRONLY))
127  {
128  switch(cmd)
129  {
130  case PIOCATTRLOOKUP:
131  case PIOCATTRREAD:
132  case PIOCINVINODE:
133  break;
134  default:
135  return -EACCES;
136  }
137  }
138  if((rv=pfsFioCheckFileSlot(fileSlot))<0)
139  return rv;
140  pfsMount=fileSlot->clink->pfsMount;
141 
142  switch(cmd)
143  {
144  case PIOCALLOC:
145  rv=pfsAllocZones(fileSlot->clink, *(int *)(arg), 1);
146  break;
147 
148  case PIOCFREE:
149  pfsFreeZones(fileSlot->clink);
150  break;
151 
152  case PIOCATTRADD:
153  case PIOCATTRDEL:
154  case PIOCATTRLOOKUP:
155  case PIOCATTRREAD:
156  rv=ioctl2Attr(fileSlot->clink, cmd, arg, buf, &fileSlot->aentryOffset);
157  break;
158 
159 #ifdef PFS_IOCTL2_INC_CHECKSUM
160  case PIOCINVINODE:
161  rv=ioctl2InvalidateInode(fileSlot->clink);
162  break;
163 #endif
164 
165  default:
166  rv=-EINVAL;
167  break;
168  }
169 
170  if(pfsMount->flags & PFS_FIO_ATTR_WRITEABLE)
171  pfsCacheFlushAllDirty(pfsMount);
172  rv=pfsFioCheckForLastError(pfsMount, rv);
173  SignalSema(pfsFioSema);
174 
175  return rv;
176 }

241 static pfs_aentry_t *getAentry(pfs_cache_t *clink, char *key, char *value, int mode)
242 {
243  int kLen, fullsize;
244  pfs_aentry_t *aentry=clink->u.aentry;
245  pfs_aentry_t *aentryLast=NULL;
246  pfs_aentry_t *end;
247 
248  kLen=strlen(key);
249  fullsize=(kLen+strlen(value)+7) & ~3;
250  for(end=(pfs_aentry_t *)((u8*)aentry+1024);aentry < end; aentry=(pfs_aentry_t *)((u8*)aentry+aentry->aLen))
251  { //Other than critical errors, do nothing about the filesystem errors.
252  if(aentry->aLen & 3)
253  PFS_PRINTF(PFS_DRV_NAME" Error: aentry allocated length/4 != 0\n");
254  if(aentry->aLen < ((aentry->kLen+aentry->vLen+7) & ~3))
255  {
256  PFS_PRINTF(PFS_DRV_NAME" Panic: aentry is too small\n");
257  return NULL;
258  }
259  if(end < (pfs_aentry_t*)((u8*)aentry+aentry->aLen))
260  PFS_PRINTF(PFS_DRV_NAME" Error: aentry too big\n");
261 
262  switch(mode)
263  {
264  case PFS_AENTRY_MODE_LOOKUP:
265  if(kLen==aentry->kLen)
266  if(memcmp(key, aentry->str, kLen)==0)
267  return aentry;
268  break;
269 
270  case PFS_AENTRY_MODE_ADD:
271  if(aentry->kLen==0)
272  {
273  if(aentry->aLen>=fullsize)
274  return aentry;
275  }
276  if(aentry->aLen - ((aentry->kLen+aentry->vLen+7) & ~3) < fullsize)
277  continue;
278  return aentry;
279 
280  case PFS_AENTRY_MODE_DELETE:
281  default:
282  if(kLen==aentry->kLen)
283  {
284  if(memcmp(key, aentry->str, kLen)==0)
285  {
286  if(aentryLast!=NULL)
287  {
288  aentryLast->aLen+=aentry->aLen;
289  return aentry;
290  }
291  // delete it :P
292  aentry->kLen=0;
293  aentry->vLen=0;
294  return aentry;
295  }
296  }
297  aentryLast=aentry;
298  break;
299 
300  }
301  }
302  return NULL;
303 }

305 static int ioctl2AttrAdd(pfs_cache_t *clink, pfs_ioctl2attr_t *attr)
306 {
307  u32 kLen, vLen;
308  pfs_aentry_t *aentry;
309  u32 tmp;
310 
311  // input check
312  kLen=strlen(attr->key);
313  vLen=strlen(attr->value);
314  if(kLen>=PFS_AENTRY_KEY_MAX || vLen>=PFS_AENTRY_VALUE_MAX) // max size bounds check
315  return -EINVAL;
316 
317  if(kLen==0 || vLen==0) // no input check
318  return -EINVAL;
319 
320  if(getAentry(clink, attr->key, NULL, PFS_AENTRY_MODE_LOOKUP))
321  return -EEXIST;
322  if((aentry=getAentry(clink, attr->key, attr->value, PFS_AENTRY_MODE_ADD)) == NULL)
323  return -ENOSPC;
324 
325  if(aentry->kLen==0)
326  tmp=aentry->aLen;
327  else
328  tmp=aentry->aLen-((aentry->kLen+(aentry->vLen + 7)) & 0x3FC); //The only case that uses 0x3FC within the whole PFS driver.
329 
330  aentry->aLen-=tmp;
331  aentry = (pfs_aentry_t*)((u8 *)aentry + aentry->aLen);
332  aentry->kLen=kLen;
333  aentry->vLen=vLen;
334  aentry->aLen=tmp;
335  memcpy(&aentry->str[0], attr->key, aentry->kLen);
336  memcpy(&aentry->str[aentry->kLen], attr->value, aentry->vLen);
337  clink->flags|=PFS_CACHE_FLAG_DIRTY;
338 
339  return 0;
340 }
