112 int pfsFsckStat(pfs_mount_t *pfsMount, pfs_super_block_t *superblock,
113  u32 stat, int mode)
114 { // mode 0=set flag, 1=remove flag, else check stat
115 
116  if(pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1,
117  PFS_IO_MODE_READ)==0)
118  {
119  switch(mode)
120  {
121  case PFS_MODE_SET_FLAG:
122  superblock->pfsFsckStat|=stat;
123  break;
124  case PFS_MODE_REMOVE_FLAG:
125  superblock->pfsFsckStat&=~stat;
126  break;
127  default/*PFS_MODE_CHECK_FLAG*/:
128  return 0 < (superblock->pfsFsckStat & stat);
129  }
130  pfsMount->blockDev->transfer(pfsMount->fd, superblock, 0, PFS_SUPER_SECTOR, 1,
131  PFS_IO_MODE_WRITE);
132  pfsMount->blockDev->flushCache(pfsMount->fd);
133  }
134  return 0;
135 }

196 pfs_block_device_t *pfsGetBlockDeviceTable(const char *name)
197 {
198  char *end;
199  char devname[32];
200  char *tmp;
201  u32 len;
202  int i;
203 
204  while(name[0] == ' ')
205  name++;
206 
207  end = strchr(name, ':');
208  if(!end) {
209  PFS_PRINTF(PFS_DRV_NAME": Error: Unknown block device '%s'\n", name);
210  return NULL;
211  }
212 
213  len = (u32)end - (u32)name;
214  strncpy(devname, name, len);
215  devname[len] = '\0';
216 
217  // Loop until digit is found, then terminate string at that digit.
218  // Should then have just the device name left, minus any front spaces or trailing digits.
219  tmp = devname;
220  while(!(isdigit(tmp[0])))
221  tmp++;
222  tmp[0] = '\0';
223 
224  for(i = 0; i < NUM_SUPPORTED_DEVICES; i++)
225  if(!strcmp(pfsBlockDeviceCallTable[i].devName, devname))
226  return &pfsBlockDeviceCallTable[i];
227 
228  return NULL;
229 }
