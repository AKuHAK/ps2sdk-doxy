25 int mcman_format1(int port, int slot)
26 {
27  register int r, i;
28  register MCDevInfo *mcdi = &mcman_devinfos[port][slot];
29 
30 #ifdef DEBUG
31  DPRINTF("mcman: mcman_format1 port%d slot%d\n", port, slot);
32 #endif
33 
34  mcman_invhandles(port, slot);
35 
36  for (i = 0; i < 56; i++) {
37  r = mcman_PS1pagetest(port, slot, i);
38  if (r == 0)
39  return sceMcResNoFormat;
40  if (r != 1)
41  return -41;
42  }
43 
44  memset(&mcman_PS1PDApagebuf, 0, 128);
45 
46  mcman_PS1PDApagebuf.word[0] = 0xa0;
47  mcman_PS1PDApagebuf.half[2] = 0xffff;
48 
49  mcman_PS1PDApagebuf.byte[127] = mcman_calcEDC(&mcman_PS1PDApagebuf, 127);

51  for (i = 1; i < 15; i++) {
52  r = McWritePS1PDACard(port, slot, i, &mcman_PS1PDApagebuf);
53  if (r != sceMcResSucceed)
54  return -42;
55  }
56 
57  memset(&mcman_PS1PDApagebuf, 0, 128);
58 
59  mcman_PS1PDApagebuf.word[0] = 0xffffffff;
60 
61  for (i = 0; i < 20; i++) {
62  mcman_PS1PDApagebuf.word[0] = mcdi->bad_block_list[i];
63  mcman_PS1PDApagebuf.byte[127] = mcman_calcEDC(&mcman_PS1PDApagebuf, 127);
64 
65  r = McWritePS1PDACard(port, slot, i + 16, &mcman_PS1PDApagebuf);
66  if (r != sceMcResSucceed)
67  return -43;
68  }
69 
70  mcman_wmemset(&mcman_PS1PDApagebuf, 128, 0);
71 
72  mcman_PS1PDApagebuf.byte[0] = 0x4d;
73  mcman_PS1PDApagebuf.byte[1] = 0x43;
74  mcman_PS1PDApagebuf.byte[127] = 0x0e;
75 
76  r = McWritePS1PDACard(port, slot, 0, &mcman_PS1PDApagebuf);
77  if (r != sceMcResSucceed)
78  return -44;
79 
80  mcdi->cardform = 1;
81

239 int mcman_read1(int fd, void *buffer, int nbyte)
240 {
241  register int r, size, temp, rpos, offset, maxsize;
242  register MC_FHANDLE *fh = (MC_FHANDLE *)&mcman_fdhandles[fd];
243  McCacheEntry *mce;
244  u8 *p = (u8 *)buffer;
245 
246  if (fh->position >= fh->filesize)
247  return 0;
248 
249  if (nbyte >= (fh->filesize - fh->position))
250  nbyte = fh->filesize - fh->position;
251 
252  rpos = 0;
253  if (nbyte) {
254  do {
255  if (fh->position < 0)
256  temp = fh->position + 0x3ff;
257  else
258  temp = fh->position;
259 
260  offset = (fh->position - ((temp >> 10) << 10));
261  maxsize = MCMAN_CLUSTERSIZE - offset;
262  if (maxsize < nbyte)
263  size = maxsize;
264  else
265  size = nbyte;
266 
267  r = mcman_fatRseekPS1(fd);
268  if (r < 0)
269  return r;
270 
271  r = mcman_readclusterPS1(fh->port, fh->slot, r, &mce);
272  if (r != sceMcResSucceed)
273  return r;
274 
275  memcpy(&p[rpos], (void *)(mce->cl_data + offset), size);
276 
277  rpos += size;
278  nbyte -= size;
279  fh->position += size;
280 
281  } while (nbyte);
282  }
283 
284  return rpos;
285 }

644 int mcman_close1(int fd)
645 {
646  register int r, temp;
647  MC_FHANDLE *fh = (MC_FHANDLE *)&mcman_fdhandles[fd];
648  McFsEntryPS1 *fse;
649  McCacheEntry *mce;
650 
651 #ifdef DEBUG
652  DPRINTF("mcman: mcman_close1 fd %d\n", fd);
653 #endif
654 
655  r = mcman_readdirentryPS1(fh->port, fh->slot, fh->freeclink, &fse);
656  if (r != sceMcResSucceed)
657  return -31;
658 
659  mce = mcman_get1stcacheEntp();
660 
661  if (fh->freeclink + 1 < 0)
662  temp = fh->freeclink + 8;
663  else
664  temp = fh->freeclink + 1;
665 
666  mce->wr_flag |= 1 << ((fh->freeclink + 1) - ((temp >> 3) << 3));
667 
668  if (fh->filesize == 0) {
669  fse->length = 0x2000;
670  }
671  else {
672  if ((fh->filesize - 1) < 0)
673  temp = (fh->filesize + 8190) >> 13;
674  else
675  temp = (fh->filesize - 1) >> 13;
676 
677  temp++;
678  fse->length = temp << 13;
679  }
680 
681  if (sio2man_type >= XSIO2MAN) {
682  fse->field_7d = 0; // <--- To preserve for XMCMAN
683  fse->field_2c = 0; //
684  fse->field_38 = 0; //
685  memset((void *)&fse->created, 0, 8); //
686  memset((void *)&fse->modified, 0, 8); //
687  }
688  else { // MCMAN does as following
689  fse->field_7d = 1;
690  fse->field_38 = fh->filesize;
691  mcman_getmcrtime(&fse->modified);
692  }
693 
694  fse->edc = mcman_calcEDC((void *)fse, 127);
695 
696  r = McFlushCache(fh->port, fh->slot);
697  if (r != sceMcResSucceed)
698  return r;
699 
700  return sceMcResSucceed;
701 }
