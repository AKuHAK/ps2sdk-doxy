732

737

739 
740  if (strcasecmp(tocEntry->filename, filename) == 0) {
741  // and it matches !!
742  return TRUE;
743  }
744  } // end of for loop
745

747  if (cacheInfoDir.cache_size == cacheInfoDir.sector_num)
748  return FALSE;

755  } else {

757  if (!cdfs_cacheDir(pathname, CACHE_START))

759  }
760  } else {
761  DPRINTF("Trying to cache directory\n\n");
762  // The wanted directory wasnt already cached, so cache it now
763  if (!cdfs_cacheDir(pathname, CACHE_START)) {
764  DPRINTF("Failed to cache directory\n\n");
765  return FALSE;
766  }
767  }
768 
769  // If we've got here, then we have a block of the directory cached, and want to check
770  // from this point, to the end of the dir
771  DPRINTF("cache_size = %d\n", cacheInfoDir.cache_size);
772  while (cacheInfoDir.cache_size > 0) {
773  tocEntryPointer = (struct DirTocEntry *)cacheInfoDir.cache;
774

776  tocEntryPointer = (struct DirTocEntry *)((u8 *)tocEntryPointer + tocEntryPointer->length);
777

779  if (tocEntryPointer->length == 0) {

781  DPRINTF("Offset into cache = %d bytes\n", (char *)tocEntryPointer - cacheInfoDir.cache);
782  tocEntryPointer = (struct DirTocEntry *)(cacheInfoDir.cache + (((((char *)tocEntryPointer - cacheInfoDir.cache) / 2048) + 1) * 2048));
783  }
784 
785  if (tocEntryPointer >= (struct DirTocEntry *)(cacheInfoDir.cache + (cacheInfoDir.cache_size * 2048))) {
786  // reached the end of the cache block
787  break;
788  }
789 
790  copyToTocEntry(tocEntry, tocEntryPointer);
791 
792  if (strcasecmp(tocEntry->filename, filename) == 0) {
793  DPRINTF("Found a matching file\n\n");
794  // and it matches !!
795  return TRUE;
796  }
797 
798  DPRINTF("Non-matching file - looking for %s , found %s\n", filename, tocEntry->filename);
799  } // end of for loop
800 
801  DPRINTF("Reached end of cache block\n");
802  // cache the next block
803  cdfs_cacheDir(pathname, CACHE_NEXT);
804  }

815 int cdfs_readSect(u32 lsn, u32 sectors, u8 *buf) {
816  u32 i;
817  u32 consecutive_sectors = (sectors > 2) ? (sectors * 2048) / 2064 : 0;
818  int retry;
819  int result = 0;
820  cdReadMode.trycount = 32;
821 
822  for (retry = 0; retry < 32; retry++) { // 32 retries
823  if (retry <= 8)
824  cdReadMode.spindlctrl = 1; // Try fast reads for first 8 tries
825  else
826  cdReadMode.spindlctrl = 0; // Then try slow reads
827 
828  if (!isValidDisc())
829  return FALSE;
830 
831  sceCdDiskReady(0);
832 
833  if (sceCdGetDiskType() == SCECdDVDV)
834  {
835  if (consecutive_sectors > 0)
836  {
837  result = !sceCdReadDVDV(lsn, consecutive_sectors, buf, &cdReadMode);
838  if (result == 0)
839  {
840  sceCdSync(0);
841  result = sceCdGetError();
842  }
843  if (result != 0)
844  {
845  continue;
846  }
847  }
848  for (i = 0; i < consecutive_sectors; i += 1)
849  {
850  memmove(buf + (2048 * i), buf + (2064 * i) + 12, 2048);
851  }
852  for (i = consecutive_sectors; i < sectors; i += 1)
853  {
854  result = !sceCdReadDVDV(lsn + i, 1, dvdvBuffer, &cdReadMode);
855  if (result == 0)
856  {
857  sceCdSync(0);
858  result = sceCdGetError();
859  }
860  if (result != 0)
861  {
862  break;
863  }
864  memcpy(buf + (2048 * i), dvdvBuffer + 12, 2048);
865  }
866  }
867  else
868  {
869  result = !sceCdRead(lsn, sectors, buf, &cdReadMode);
870  if (result == 0)
871  {
872  sceCdSync(0);
873  result = sceCdGetError();
874  }
875  }
876 
877  if (result == 0)
878  break;
879  }
880 
881  cdReadMode.trycount = 32;
882  cdReadMode.spindlctrl = 1;
883 
884  if (result == 0)
885  return TRUE;
886 
887  memset(buf, 0, (sectors << 11));
888 
889  return FALSE; // error
890 }
