164 static int audsrv_adpcm_alloc_channel(void)
165 {

208 int audsrv_ch_play_adpcm(int ch, u32 id)
209 {
210  int channel;
211  u32 endx;
212  adpcm_list_t *a;
213 
214  a = adpcm_loaded(id);
215  if (a == NULL)
216  {
217  /* bad joke */
218  return AUDSRV_ERR_ARGS;
219  }
220 
221  /* sample was loaded */
222  if (ch >= 0 && ch < 24)
223  {
224  endx = sceSdGetSwitch(SD_CORE_1 | SD_SWITCH_ENDX);
225  if (!(endx & (1 << ch)))
226  {
227  /* Channel in use. */
228  return -AUDSRV_ERR_NO_MORE_CHANNELS;
229  }
230 
231  channel = ch;
232  }
233  else
234  {
235  channel = audsrv_adpcm_alloc_channel();
236  if (channel < 0)
237  return channel;
238  }
239 
240  sceSdSetParam(SD_CORE_1 | (channel << 1) | SD_VPARAM_PITCH, a->pitch);
241  sceSdSetAddr(SD_CORE_1 | (channel << 1) | SD_VOICE_START, a->spu2_addr);
242  sceSdSetSwitch(SD_CORE_1 | SD_SWITCH_KON, (1 << channel));
243  return AUDSRV_ERR_NOERROR;
244 }
