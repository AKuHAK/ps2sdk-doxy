28 static void UpdatePadThread(void *arg)
29 {
30  iop_thread_info_t tstatus;
31  padState_t *pstate;
32  u32 res = 0;
33 
34  ReferThreadStatus(TH_SELF, &tstatus);
35  pstate = (padState_t*)tstatus.option;
36 
37  while(1)
38  {
39  WaitClearEvent(pstate->eventflag, EF_UPDATE_PAD, WEF_AND|WEF_CLEAR, NULL);
40 
41  // Check if able to read button data from pad
42  if( ReadData(pstate) == 1)
43  {
44  res = 0;
45 
46  if( (pstate->outbuffer[2] != 0x5a) || ( pstate->outbuffer[1] != pstate->modeCurId) || (pstate->outbuffer[1] == 0xf3) )
47  {
48  pstate->buttonDataReady = 0;
49  pstate->state = PAD_STATE_EXECCMD ;
50  pstate->currentTask = TASK_QUERY_PAD;
51  StartThread(pstate->querypadTid, NULL);
52  }
53  else
54  {
55  u32 i;
56 
57  for(i=0; i < 32; i++)
58  pstate->buttonStatus[i] = pstate->outbuffer[i];
59 
60  pstate->buttonDataReady = 1;
61 
62  if(pstate->modeConfig == MODE_CONFIG_QUERY_PAD)
63  pstate->state = PAD_STATE_FINDCTP1;
64  else
65  pstate->state = PAD_STATE_STABLE ;
66 
67  if( (pstate->reqState == PAD_RSTAT_BUSY) && (pstate->runTask == TASK_NONE))
68  pstate->reqState = PAD_RSTAT_COMPLETE;
69  }
70  }
71  else
72  {
73  pstate->buttonDataReady = 0;
74  pstate->state = PAD_STATE_ERROR;
75  pstate->findPadRetries++;
76 
77  res += pdGetError( pstate->port, pstate->slot );
78 
79  if(res >= 10)
80  {
81  pstate->currentTask = TASK_QUERY_PAD;
82  StartThread(pstate->querypadTid, NULL);
83  }
84  }
85  }
86 }

648 s32 padPortOpen(s32 port, s32 slot, s32 pad_area_ee_addr, u32 *buf)
649 {
650  iop_event_t event;
651  iop_thread_t thread;
652 
653  if(port > 2)
654  {
655  M_PRINTF("Invalid port number: %d\n", (int)port);
656  return 0;
657  }
658 
659  if(slot > 4)
660  {
661  M_PRINTF("Invalid slot number: %d\n", (int)port);
662  return 0;
663  }
664 
665  if( (openSlots[port] >> slot) & 0x1)
666  {
667  M_PRINTF("This slot is already open: (%d, %d)\n", (int)port, (int)slot);
668  return 0;
669  }
670 
671  padState[port][slot].port = port;
672  padState[port][slot].slot = slot;
673 
674  openSlots[port] |= 1 << slot;
675 
676  padState[port][slot].state = PAD_STATE_EXECCMD;
677  padState[port][slot].modeCurId = 0;
678  padState[port][slot].reqState = PAD_RSTAT_COMPLETE;
679  padState[port][slot].frame = 0;
680  padState[port][slot].padarea_ee_addr = pad_area_ee_addr;
681  padState[port][slot].buttonDataReady = 0;
682  padState[port][slot].ee_actDirectSize = 0;
683  padState[port][slot].val_c6 = 0;
684  padState[port][slot].currentTask = TASK_NONE;
685  padState[port][slot].runTask = TASK_NONE;
686  padState[port][slot].val_184 = 0;
687 
688  padState[port][slot].disconnected = 1;
689  padState[port][slot].stat70bit = 0;
690 
691  event.attr = EA_MULTI;
692  event.bits = 0;
693 
694  if((padState[port][slot].eventflag = CreateEventFlag(&event)) == 0)
695  {
696  M_PRINTF("Port open failed (CreateEventFlag).\n");
697  return 0;
698  }
699 
700 
701  // Create UpdatePadThread
702  thread.attr = TH_C;
703  thread.option = (u32)&padState[port][slot];
704  thread.thread = &UpdatePadThread;
705  thread.stacksize = 0x600;
706  thread.priority = thpri_lo;
707
