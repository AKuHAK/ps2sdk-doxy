196 void* poweroff_rpc_server(int fno, void *data, int size)
197 {
198  switch(fno) {

200  Shutdown(0);
201  break;
202 
203  case PWROFF_ENABLE_AUTO_SHUTOFF:
204  InitPowerOffThread();

206  int* sbuff = data;
207  if (sbuff[0])
208  SetPowerButtonHandler(Shutdown, 0);
209  else
210  SetPowerButtonHandler(SendCmd, 0);

213  }
214  return NULL;
