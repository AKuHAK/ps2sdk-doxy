30 int apaCacheInit(u32 size)
31 {
32  apa_header_t *header;
33  unsigned int i;
34 
35  cacheSize=size; // save size ;)
36  if((header=(apa_header_t *)apaAllocMem(size*sizeof(apa_header_t)))){
37  cacheBuf=apaAllocMem((size+1)*sizeof(apa_cache_t));
38  if(cacheBuf==NULL)
39  return -ENOMEM;
40  }
41  else
42  return -ENOMEM;
43  // setup cache header...
44  memset(cacheBuf, 0, (size+1)*sizeof(apa_cache_t));
45  cacheBuf->next=cacheBuf;
46  cacheBuf->tail=cacheBuf;
47  for(i=1; i<size+1;i++, header++){
48  cacheBuf[i].header=header;
49  cacheBuf[i].device=-1;
50  apaCacheLink(cacheBuf->tail, &cacheBuf[i]);
51  }
52  return 0;
53 }
