116  // Get mcman lib ptr

233 void *cb_rpc_S_0400(u32 fno, void *buf, int size)
234 {
235  // Rpc Callback function
236  int (*rpc_func)(void);
237  register int i;
238 
239  if (mcman_type == XMCMAN) {
240  for (i=0; i<16; i++) { // retrieve correct function number for xmcserv
241  if (fno == XMCSERV_RpcCmd[0][i]) {
242  fno = XMCSERV_RpcCmd[1][i];
243  break;
244  }
245  }
246  }
247  else { // retrieve correct function number for mcserv
248  fno -= 112;
249  if (fno > 16)
250  return (void *)&rpc_stat;
251  }
252 
253  // Get function pointer
254  rpc_func = (void *)rpc_funcs_array[fno];
255 
256  // Call needed rpc func
257  rpc_stat.result = rpc_func();
258 
259  pMcReplaceBadBlock();
260 
261  return (void *)&rpc_stat;
262 }

458 int _McRead(void *rpc_buf)
459 {
460  mcDescParam_t *dP = (mcDescParam_t *)rpc_buf;
461  register int file_offset, status;
462  register int left_to_read, size_readed, size_to_read;
463  mcEndParam_t eP;
464  SifDmaTransfer_t dmaStruct;
465  int intStatus;
466  void *eedata;
467 
468 #ifdef DEBUG
469  DPRINTF("mcserv: _McRead fd %d ee buffer addr %x size %d\n", dP->fd, (int)dP->buffer, dP->size);
470 #endif
471 
472  eP.size1 = 0;
473  eP.size2 = 0;
474  eP.dest1 = NULL;
475  eP.dest2 = NULL;
476 
477  size_readed = 0;
478  file_offset = 0;
479 
480  eedata = dP->buffer;
481 
482  eP.size1 = dP->size;
483 
484  if (dP->size > 16)
485  eP.size1 = (((int)(eedata)-1) & 0xfffffff0) - ((int)(eedata) - 16);
486 
487  eP.size2 = (dP->size - eP.size1) & 0x0f;
488  left_to_read = (dP->size - eP.size1) - eP.size2;
489 
490  if (eP.size2 != 0)
491  eP.dest2 = (void *)(eedata + eP.size1 + left_to_read);
492 
493  if (eP.size1 != 0) {
494  size_readed = pMcRead(dP->fd, eP.src1, eP.size1);
495 
496  if (size_readed < 0) {
497  eP.size1 = 0;
498  eP.size2 = 0;
499  goto dma_transfer2;
500  }
501  else {
502  file_offset = size_readed;
503  eP.dest1 = eedata;
504  eedata += size_readed;
505 
506  if (size_readed != eP.size1) {
507  eP.size1 = size_readed;
508  eP.size2 = 0;
509  size_readed = 0;
510  goto dma_transfer2;
511  }
512  else
513  size_readed = 0;
514  }
515  }
516 
517  while (left_to_read > 0) {
518 
519  size_to_read = left_to_read;
520 
521  if (left_to_read > MCSERV_BUFSIZE)
522  size_to_read = MCSERV_BUFSIZE;
523 
524  size_readed = pMcRead(dP->fd, mcserv_buf, size_to_read);
525 
526  if (size_readed < 0) {
527  eP.size2 = 0;
528  goto dma_transfer2;
529  }
530 
531  dmaStruct.src = (void *)mcserv_buf;
532  dmaStruct.dest = (void *)eedata;
533  dmaStruct.size = size_readed;
534  dmaStruct.attr = 0;
535 
536  CpuSuspendIntr(&intStatus);
537  sceSifSetDma(&dmaStruct, 1);
538 
539  file_offset += size_readed;
540  left_to_read -= size_readed;
541 
542  CpuResumeIntr(intStatus);
543 
544  eedata += size_readed;
545 
546  if (size_to_read != size_readed) {
547  eP.size2 = 0;
548  size_readed = 0;
549  goto dma_transfer2;
550  }
551  size_readed = 0;
552  }
553 
554  if (eP.size2 == 0)
555  goto dma_transfer2;
556 
557  size_readed = pMcRead(dP->fd, eP.src2, eP.size2);
558 
559  if (size_readed < 0) {
560  eP.size2 = 0;
561  goto dma_transfer2;
562  }
563 
564  file_offset += size_readed;
565  eP.size2 = size_readed;
566  size_readed = 0;
567 
568 dma_transfer2:
569  dmaStruct.src = (void *)&eP;
570  dmaStruct.dest = (void *)dP->param;
571  dmaStruct.size = sizeof (mcEndParam_t);
572  dmaStruct.attr = 0;
573 
574  CpuSuspendIntr(&intStatus);
575  status = sceSifSetDma(&dmaStruct, 1);
576  CpuResumeIntr(intStatus);
577 
578  while (sceSifDmaStat(status) >= 0)
579  DelayThread(100);
580 
581  if (size_readed != 0)
582  return size_readed;
583 
584  return file_offset;
585 }

616

618  left_to_read = (dP->size - eP.size1) - eP.size2;

674

676 
677 dma_transfer:
