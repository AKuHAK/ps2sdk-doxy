141 //-------------------------------------------------------------------------
142 /* RecvTimeout() is used instead of just recv(), in order to prevent
143  lockups in the event of a network failure in-between calls to socket functions.
144 
145  TCP only protects against failures that occur during calls to socket functions.
146 
147  Just to illustrate its importance here, imagine what would happen if recv() is called,
148  right before the server terminates the connection without the PlayStation 2 ever
149  receiving the TCP retransmission and the TCP RST packets. */
150 static int RecvTimeout(int sock, void *buf, int bsize, int timeout_ms)
151 {
152  int ret;
153  struct pollfd pollfd[1];
154 
155  pollfd->fd = sock;
156  pollfd->events = POLLIN;
157  pollfd->revents = 0;
158 
159  ret = poll(pollfd, 1, timeout_ms);
160 
161  // a result less than 0 is an error
162  if (ret < 0)
163  return -1;
164 
165  // 0 is a timeout
166  if (ret == 0)
167  return 0;
168 
169  // receive the packet
170  ret = lwip_recv(sock, buf, bsize, 0);
171  if (ret < 0)
172  return -2;
173 
174  return ret;
175 }

177 static int SendData(int sock, char *buf, int size)
178 {
179  int remaining, result;
180  char *ptr;
181 
182  ptr = buf;
183  remaining = size;
184  while (remaining > 0)
185  {
186  result = lwip_send(sock, ptr, remaining, 0);
187  if (result <= 0)
188  return result;
189 
190  ptr += result;
191  remaining -= result;
192  }
193 
194  return size;
195 }

198 {
199  int remaining, result;
200  char *ptr;

202  ptr = buf;
203  remaining = size;
204  while (remaining > 0)

206  result = RecvTimeout(sock, ptr, remaining, timeout_ms);
207  if (result <= 0)

209 
210  ptr += result;
211  remaining -= result;
212  }
213 
214  return size;

217 //-------------------------------------------------------------------------
218 static int GetSMBServerReply(int shdrlen, void *spayload, int rhdrlen)
219 {
220  int rcv_size, totalpkt_size, size;
221 
222  totalpkt_size = nb_GetSessionMessageLength() + 4;
223 
224  if (shdrlen == 0)
225  {
226  //Send the whole message, including the 4-byte direct transport packet header.
227  rcv_size = SendData(main_socket, (char*)&SMB_buf, totalpkt_size);
228  if (rcv_size <= 0)
229  return -1;
230  }
231  else
232  {
233  size = shdrlen + 4;
234 
235  //Send the headers, followed by the payload.
236  rcv_size = SendData(main_socket, (char*)&SMB_buf, size);
237  if (rcv_size <= 0)
238  return -1;
239 
240  rcv_size = SendData(main_socket, spayload, totalpkt_size - size);
241  if (rcv_size <= 0)
242  return -1;
243  }
244 
245  //Read NetBIOS session message header. Drop NBSS Session Keep alive messages (type == 0x85, with no body), but process session messages (type == 0x00).
246  do{
247  rcv_size = RecvData(main_socket, (char *)&SMB_buf.sessionHeader, sizeof(SMB_buf.sessionHeader), 10000); // 10s before the packet is considered lost
248  if (rcv_size <= 0)
249  return -2;
250  } while (nb_GetPacketType() != 0);
251 
252  totalpkt_size = nb_GetSessionMessageLength();
253 
254  //If rhdrlen is not specified, retrieve the whole packet. Otherwise, retrieve only the headers (caller will retrieve the payload separately).
255  size = (rhdrlen == 0) ? totalpkt_size : rhdrlen;
256  rcv_size = RecvData(main_socket, (char *)&SMB_buf.smb, size, 3000); // 3s before the packet is considered lost
257  if (rcv_size <= 0)
258  return -2;
259 
260  return totalpkt_size;
261 }

264 //These functions will process UTF-16 characters on a byte-level, so that they will be safe for use with byte-alignment.
265 static int asciiToUtf16(char *out, const char *in)
266 {
267  int len;
268  const char *pIn;
269  u8 *pOut;
270 
271  for(pIn = in, pOut = out, len = 0; *pIn != '\0'; pIn++, pOut += 2, len += 2)
272  {
273  pOut[0] = *pIn;
274  pOut[1] = '\0';
275  }
276 
277  pOut[0] = '\0'; //NULL terminate.
278  pOut[1] = '\0';
279  len += 2;
280 
281  return len;
282 }

284 static int utf16ToAscii(char *out, const char *in, int inbytes)
285 {
286  int len, bytesProcessed;
287  const u8 *pIn;
288  char *pOut;
289  u16 wchar;
290 
291  for(pIn = in, pOut = out, len = 0, bytesProcessed = 0; (inbytes == 0) || (bytesProcessed < inbytes) ; len++)
292  {
293  wchar = pIn[0] | ((u16)pIn[1] << 8);
294  if (wchar == '\0')
295  break;
296 
297  if (wchar >= 0xD800 && wchar < 0xDC00)
298  { //Skip surrogate. Replace unsupported character with '?'.
299  *pOut = '?';
300 
301  pIn += 4;
302  bytesProcessed += 4;
303  pOut++;
304  } else {
305  // Write decoded character. Replace unsupported characters with '?'.
306  *pOut = (wchar > 128) ? '?' : (char)wchar;
307 
308  pIn += 2;
309  bytesProcessed += 2;
310  pOut++;
311  }
312  }
313 
314  pOut[0] = '\0'; //NULL terminate.
315  len++;
316 
317  return len;
318 }

424 static int AddPassword(char *Password, int PasswordType, int AuthType, u16 *AnsiPassLen, u16 *UnicodePassLen, u8 *Buffer)
425 {
426  u8 passwordhash[16];
427  u8 LMresponse[24];
428  u16 passwordlen = 0;
429 
430  if ((Password) && (PasswordType != NO_PASSWORD)) {
431  if (server_specs.PasswordType == SERVER_USE_ENCRYPTED_PASSWORD) {
432  passwordlen = 24;
433  switch (PasswordType) {
434  case HASHED_PASSWORD:
435  if (AuthType == LM_AUTH) {
436  memcpy(passwordhash, Password, 16);
437  memcpy(AnsiPassLen, &passwordlen, 2);
438  }
439  if (AuthType == NTLM_AUTH) {
440  memcpy(passwordhash, &Password[16], 16);
441  memcpy(UnicodePassLen, &passwordlen, 2);
442  }
443  break;
444 
445  default:
446  if (AuthType == LM_AUTH) {
447  LM_Password_Hash((const unsigned char*)Password, passwordhash);
448  memcpy(AnsiPassLen, &passwordlen, 2);
449  }
450  else if (AuthType == NTLM_AUTH) {
451  NTLM_Password_Hash((const unsigned char*)Password, passwordhash);
452  memcpy(UnicodePassLen, &passwordlen, 2);
453  }
454  }
455  LM_Response(passwordhash, server_specs.EncryptionKey, LMresponse);
456  memcpy(Buffer, LMresponse, passwordlen);
457  }
458  else if (server_specs.PasswordType == SERVER_USE_PLAINTEXT_PASSWORD) {
459  // It seems that PlainText passwords and Unicode isn't meant to be...
460  passwordlen = strlen(Password);
461  if (passwordlen > 14)
462  passwordlen = 14;
463  else if (passwordlen == 0)
464  passwordlen = 1;
465  memcpy(AnsiPassLen, &passwordlen, 2);
466  memcpy(Buffer, Password, passwordlen);
467  }
468  }
469  else {
470  if (server_specs.SecurityMode == SERVER_SHARE_SECURITY_LEVEL) {
471  passwordlen = 1;
472  memcpy(AnsiPassLen, &passwordlen, 2);
473  Buffer[0] = 0;
474  }
475  }
476 
477  return passwordlen;
478 }

484  SessionSetupAndXResponse_t *SSRsp = &SMB_buf.smb.sessionSetupAndXResponse;

486  int passwordlen = 0;

488 
489 lbl_session_setup:
490 
491  ZERO_PKT_ALIGNED(SSR, sizeof(SessionSetupAndXRequest_t));
492 
493  useUnicode = (server_specs.Capabilities & SERVER_CAP_UNICODE) ? 1 : 0;
494 
495  SSR->smbH.Magic = SMB_MAGIC;
496  SSR->smbH.Cmd = SMB_COM_SESSION_SETUP_ANDX;
497  SSR->smbH.Flags = SMB_FLAGS_CASELESS_PATHNAMES;
498  SSR->smbH.Flags2 = SMB_FLAGS2_KNOWS_LONG_NAMES | SMB_FLAGS2_32BIT_STATUS;
499  if (useUnicode)
500  SSR->smbH.Flags2 |= SMB_FLAGS2_UNICODE_STRING;
501  SSR->smbWordcount = 13;
502  SSR->smbAndxCmd = SMB_COM_NONE; // no ANDX command
503  SSR->MaxBufferSize = CLIENT_MAX_BUFFER_SIZE;
504  SSR->MaxMpxCount = server_specs.MaxMpxCount >= 2 ? 2 : (u16)server_specs.MaxMpxCount;
505  SSR->VCNumber = 1;
506  SSR->SessionKey = server_specs.SessionKey;
507  SSR->Capabilities = capabilities;
508 
509  // Fill ByteField
510  offset = 0;
511 
512  if (server_specs.SecurityMode == SERVER_USER_SECURITY_LEVEL) {
513  passwordlen = AddPassword(Password, PasswordType, AuthType, &SSR->AnsiPasswordLength, &SSR->UnicodePasswordLength, &SSR->ByteField[0]);
514  offset += passwordlen;
515  }
516 
517  if ((useUnicode) && (!(passwordlen & 1)))
518  {
519  SSR->ByteField[offset] = '\0';
520  offset++; // pad needed only for unicode as aligment fix if password length is even
521  }
522 
523  // Add User name
524  offset += setStringField(&SSR->ByteField[offset], User);
525 
526  // PrimaryDomain, acquired from Negotiate Protocol Response data
527  offset += setStringField(&SSR->ByteField[offset], server_specs.PrimaryDomainServerName);
528 
529  // NativeOS
530  if (useUnicode && ((offset & 1) != 0))
531  { //If Unicode is used, the field must begin on a 16-bit aligned address.
532  SSR->ByteField[offset] = '\0';
533  offset++;
534  }
535  offset += setStringField(&SSR->ByteField[offset], "PlayStation 2");
536 
537  // NativeLanMan
538  if (useUnicode && ((offset & 1) != 0))
539  { //If Unicode is used, the field must begin on a 16-bit aligned address.
540  SSR->ByteField[offset] = '\0';
541  offset++;
542  }
543  offset += setStringField(&SSR->ByteField[offset], "SMBMAN");
544 
545  SSR->ByteCount = offset;
546 
547  nb_SetSessionMessage(sizeof(SessionSetupAndXRequest_t)+offset);
548  r = GetSMBServerReply(0, NULL, 0);
549  if (r <= 0)
550  return -EIO;
551 
552  // check sanity of SMB header
553  if (SSRsp->smbH.Magic != SMB_MAGIC)
554  return -EIO;
555 
556  // check there's no error (NT STATUS error type!)
557  switch(SSRsp->smbH.Eclass | (SSRsp->smbH.Ecode << 16)){
558  case STATUS_SUCCESS:
559  break;
560  case STATUS_LOGON_FAILURE: // check there's no auth failure
561  if ((server_specs.SecurityMode == SERVER_USER_SECURITY_LEVEL)
562  && (AuthType == NTLM_AUTH)) {
563  AuthType = LM_AUTH;
564  goto lbl_session_setup;
565  }
566 
567  return -EACCES;
568  default:
569  return -EIO;
570  }
571 
572  // return UID
573  return (int)SSRsp->smbH.UID;
574 }
