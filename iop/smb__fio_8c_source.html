106 // Timer Interrupt handler for Echoing the server every 3 seconds, when
107 // not already doing IO, and counting after an IO operation has finished
108 //
109 static unsigned int timer_intr_handler(void *args)
110 {
111  iSignalSema(keepalive_mutex);
112  iSetAlarm(&keepalive_sysclock, timer_intr_handler, NULL);
113 
114  return (unsigned int)args;
115 }

120  int oldstate;

170 //-------------------------------------------------------------------------
171 static void keepalive_thread(void *args)
172 {
173  int r, opened_share = 0;
174 
175  while (1) {
176  // wait for keepalive mutex
177  WaitSema(keepalive_mutex);
178 
179  // ensure no IO is already processing
180  WaitSema(smbman_io_sema);
181 
182  // echo the SMB server
183  r = smb_Echo("PS2 KEEPALIVE ECHO", 18);
184  if (r < 0) {
185  keepalive_lock();
186 
187  if (TID != -1)
188  opened_share = 1;
189 
190  if (UID != -1) {
191  r = smb_LogOn((smbLogOn_in_t *)&glogon_info);
192  if (r == 0) {
193  if (opened_share)
194  smb_OpenShare((smbOpenShare_in_t *)&gopenshare_info);
195  }
196  }
197 
198  keepalive_unlock();
199  }
200 
201  SignalSema(smbman_io_sema);
202  }
203 }

290 static char *prepare_path(const char *path, char *full_path, int max_path)
291 {
292  const char *p, *p2;
293  int i;
294 
295  //Reserve space for 2 backslashes and a NULL.
296  strncpy(full_path, smb_curdir, max_path - 3);
297  strcat(full_path, "\\");
298 
299  //Skip all leading slashes and backslashes.
300  p = path;
301  while ((*p == '\\') || (*p == '/'))
302  p++;
303 
304  //Locate the end of the path, ignoring any trailing slashes and backslashes.
305  p2 = &path[strlen(path)];
306  while ((*p2 == '\\') || (*p2 == '/'))
307  p2--;
308 
309  //Copy path. Reserve space for a backslash and a NULL
310  for (i = strlen(full_path); (p <= p2) && (max_path - i - 2 > 0); p++,i++)
311  { //Convert all slashes along the path to backslashes.
312  full_path[i] = (*p == '/') ? '\\' : *p;
313  }
314 
315  //Append a backslash and null-terminate.
316  full_path[i] = '\\';
317  full_path[i+1] = '\0';
318 
319  return full_path;
320 }

543 static void FileTimeToDate(u64 FileTime, u8 *datetime)
544 {
545  u8 daysPerMonth[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
546  int i;
547  u64 time;
548  u16 years, days;
549  u8 leapdays, months, hours, minutes, seconds;
550 
551  time = FileTime / 10000000; // convert to seconds from 100-nanosecond intervals
552 
553  years = (u16)(time / ((u64)60 * 60 * 24 * 365)); // hurray for interger division
554  time -= years * ((u64)60 * 60 * 24 *365); // truncate off the years
555 
556  leapdays = (years / 4) - (years / 100) + (years / 400);
557  years += 1601; // add base year from FILETIME struct;
558 
559  days = (u16)(time / (60 * 60 * 24));
560  time -= (unsigned int)days * (60 * 60 * 24);
561  days -= leapdays;
562 
563  if( (years % 4) == 0 && ((years % 100) != 0 || (years % 400) == 0) )
564  daysPerMonth[1]++;
565 
566  months = 0;
567  for( i=0; i<12; i++ )
568  {
569  if( days > daysPerMonth[i] )
570  {
571  days -= daysPerMonth[i];
572  months++;
573  }
574  else
575  break;
576  }
577 
578  if( months >= 12 )
579  {
580  months -= 12;
581  years++;
582  }
583  hours = (u8)(time / (60 * 60));
584  time -= (u16)hours * (60 * 60);
585 
586  minutes = (u8)(time / 60);
587  time -= minutes * 60;
588 
589  seconds = (u8)(time);
590 
591  datetime[0] = 0;
592  datetime[1] = seconds;
593  datetime[2] = minutes;
594  datetime[3] = hours;
595  datetime[4] = days + 1;
596  datetime[5] = months + 1;
597  datetime[6] = (u8)(years & 0xFF);
598  datetime[7] = (u8)((years >> 8) & 0xFF);
599 }
