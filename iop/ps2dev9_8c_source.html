395 {

397  volatile iop_dmac_chan_t *dev9_chan = (iop_dmac_chan_t *)DEV9_DMAC_BASE;
398  int res, dmactrl, OldState;
399 
400  if(device >= 2)
401  {
402  if (dev9_predma_cbs[device] == NULL) return -1;
403  if (dev9_postdma_cbs[device] == NULL) return -1;
404  }
405 
406  if ((res = WaitSema(dma_lock_sem)) < 0)
407  return res;
408 
409  switch (device)
410  {
411  case 0:
412  dmactrl = 0;
413  break;
414  case 1:
415  dmactrl = 1;
416  break;
417 #ifdef DEV9_PSX_SUPPORT
418  //Used for the PSX
419  case 2:
420  dmactrl = 0x10;
421  break;
422  case 3:
423  dmactrl = 0x20;
424  break;
425 #endif
426  default:
427  dmactrl = 0;
428  }
429

431

433  dev9_predma_cbs[device](bcr, dir);

447  dev9_postdma_cbs[device](bcr, dir);
448

453 static int read_eeprom_data(void)
454 {
455  USE_SPD_REGS;
456  int i, j, res = -2;
457  u8 val;
458 
459  if (eeprom_data[0] < 0)
460  goto out;
461 
462  SPD_REG8(SPD_R_PIO_DIR) = 0xe1;
463  DelayThread(1);
464  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
465  DelayThread(1);
466 
467  for (i = 0; i < 2; i++) {
468  SPD_REG8(SPD_R_PIO_DATA) = 0xa0;
469  DelayThread(1);
470  SPD_REG8(SPD_R_PIO_DATA) = 0xe0;
471  DelayThread(1);
472  }
473  for (i = 0; i < 7; i++) {
474  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
475  DelayThread(1);
476  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
477  DelayThread(1);
478  }
479  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
480  DelayThread(1);
481 
482  val = SPD_REG8(SPD_R_PIO_DATA);
483  DelayThread(1);
484  if (val & 0x10) { /* Error. */
485  SPD_REG8(SPD_R_PIO_DATA) = 0;
486  DelayThread(1);
487  res = -1;
488  eeprom_data[0] = 0;
489  goto out;
490  }
491 
492  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
493  DelayThread(1);
494 
495  /* Read the MAC address and checksum from the EEPROM. */
496  for (i = 0; i < 4; i++) {
497  eeprom_data[i+1] = 0;
498 
499  for (j = 15; j >= 0; j--) {
500  SPD_REG8(SPD_R_PIO_DATA) = 0xc0;
501  DelayThread(1);
502  val = SPD_REG8(SPD_R_PIO_DATA);
503  if (val & 0x10)
504  eeprom_data[i+1] |= (1<<j);
505  SPD_REG8(SPD_R_PIO_DATA) = 0x80;
506  DelayThread(1);
507  }
508  }
509 
510  SPD_REG8(SPD_R_PIO_DATA) = 0;
511  DelayThread(1);
512  eeprom_data[0] = 1; /* The EEPROM data is valid. */
513  res = 0;
514 
515 out:
516  SPD_REG8(SPD_R_PIO_DIR) = 1;
517  return res;
518 }
